 21def is_top(term: KInner, *, weak: bool = False) -> bool:
 22    if _is_top(term):
 23        return True
 24    if not weak:
 25        return False
 26    flat = flatten_label('#And', term)
 27    if len(flat) == 1:
 28        return is_top(single(flat))
 29    return all(is_top(term, weak=True) for term in flat)

 36def is_bottom(term: KInner, *, weak: bool = False) -> bool:
