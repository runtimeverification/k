 32    @staticmethod
 33    def from_decl(decl: SortDecl) -> Collection:
 34        if not 'element' in decl.attrs_by_key:
 35            raise ValueError(f'Not a sort declaration: {decl.text}')
 36        sort = decl.name
 37        concat = Collection._extract_label(decl, 'concat')
 38        element = Collection._extract_label(decl, 'element')
 39        unit = Collection._extract_label(decl, 'unit')
 40        kind = CollectionKind(Collection._extract_string(decl, 'hook').split('.')[1])  # 'MAP.Map' -> CollectionKind.MAP
 41        return Collection(
 42            sort=sort,
 43            concat=concat,
 44            element=element,
 45            unit=unit,
 46            kind=kind,
 47        )

 75    @staticmethod
 76    def from_definition(defn: Definition) -> KoreDefn:
 77        sorts: dict[str, SortDecl] = {}
 78        symbols: dict[str, SymbolDecl] = {}
 79        subsorts: list[tuple[str, str]] = []
 80        rewrites: list[RewriteRule] = []
 81        functions: dict[str, list[FunctionRule]] = {}
 82
 83        for sent in defn.sentences:
 84            match sent:
 85                case SortDecl(name):
 86                    sorts[name] = sent
 87                case SymbolDecl(Symbol(name)):
 88                    symbols[name] = sent
 89                    if 'function' in sent.attrs_by_key:
 90                        functions.setdefault(name, [])
 91                case Axiom(attrs=(App('subsort', (SortApp(subsort), SortApp(supersort))),)):
 92                    subsorts.append((subsort, supersort))
 93                case Axiom():
 94                    if Rule.is_rule(sent):
 95                        rule = Rule.from_axiom(sent)
 96                        if isinstance(rule, RewriteRule):
 97                            rewrites.append(rule)
 98                        elif isinstance(rule, FunctionRule):
 99                            functions.setdefault(rule.lhs.symbol, []).append(rule)
100
101        return KoreDefn(
102            sorts=FrozenDict(sorts),
103            symbols=FrozenDict(symbols),
104            subsorts=POSet((supersort, subsort) for subsort, supersort in subsorts).image,
105            rewrites=tuple(rewrites),
106            functions=FrozenDict((key, tuple(values)) for key, values in functions.items()),
107        )

110    def constructors(self) -> FrozenDict[str, tuple[str, ...]]:
111        grouped: dict[str, list[str]] = {}
112        for symbol, decl in self.symbols.items():
113            if not 'constructor' in decl.attrs_by_key:
114                continue
115            sort = check_type(decl.sort, SortApp).name  # TODO eliminate by further processing the SortDecl
116            grouped.setdefault(sort, []).append(symbol)
117        return FrozenDict((sort, tuple(symbols)) for sort, symbols in grouped.items())

119    @cached_property
120    def collections(self) -> FrozenDict[str, Collection]:
121        colls: dict[str, Collection] = {}
122        for sort, decl in self.sorts.items():
123            if not 'element' in decl.attrs_by_key:
124                continue
125            coll = Collection.from_decl(decl)
126            assert sort == coll.sort
127            colls[sort] = coll
128        return FrozenDict(colls)

152    def project_to_symbols(self) -> KoreDefn:
153        """Project definition to symbols present in the definition."""
154        functions: FrozenDict[str, tuple[FunctionRule, ...]] = FrozenDict(
155            (function, rules) for function, rules in self.functions.items() if function in self.symbols
156        )
157
158        _sorts: set[str] = set()
159        _sorts.update(sort for symbol in self.symbols for sort in self._symbol_sorts(symbol))
160        _sorts.update(sort for rules in functions.values() for rule in rules for sort in self._arg_subsorts(rule))
161        sorts: FrozenDict[str, SortDecl] = FrozenDict(
162            (sort, decl) for sort, decl in self.sorts.items() if sort in _sorts
163        )
164
165        subsorts: FrozenDict[str, frozenset[str]] = FrozenDict(
166            (supersort, frozenset(subsort for subsort in subsorts if subsort in sorts))
167            for supersort, subsorts in self.subsorts.items()
168            if supersort in sorts
169        )
170
171        return self.let(
172            sorts=sorts,
173            subsorts=subsorts,
174            functions=functions,
175        )

177    def project_to_rewrites(self) -> KoreDefn:
178        """Project definition to symbols that are part of the configuration or are (transitively) referred to from rewrite axioms."""
179        _symbols = set()
180        _symbols.update(self._config_symbols())
181        _symbols.update(self._rewrite_symbols())
182        symbols: FrozenDict[str, SymbolDecl] = FrozenDict(
183            (symbol, decl) for symbol, decl in self.symbols.items() if symbol in _symbols
184        )
185        return self.let(symbols=symbols).project_to_symbols()

187    def _symbol_sorts(self, symbol: str) -> list[str]:
188        decl = self.symbols[symbol]
189        res = []
190        if isinstance(decl.sort, SortApp):
191            res.append(decl.sort.name)
192        res.extend(sort.name for sort in decl.param_sorts if isinstance(sort, SortApp))
193        return res

195    def _arg_subsorts(self, rule: FunctionRule) -> set[str]:
196        """Collect the from-sorts of injections on the LHS of a function rule.
197
198        These potentially indicate matching on a subsort of a certain sort, e.g.
199        rule isKResult(inj{KResult,KItem}(KResult)) => true
200        """
201        res = set()
202
203        def from_sort(pattern: Pattern) -> None:
204            match pattern:
205                case App('inj', (SortApp(sort), _), _):
206                    res.add(sort)
207
208        rule.lhs.collect(from_sort)
209        return res

211    def _config_symbols(self) -> set[str]:
212        """Return the set of symbols that constitute a configuration."""
213        res: set[str] = set()
214        done = set()
215        pending = ['SortGeneratedTopCell']
216        while pending:
217            sort = pending.pop()
218            if sort in done:
219                continue
220            done.add(sort)
221
222            symbols: list[str] = []
223            if sort in self.collections:
224                coll = self.collections[sort]
225                symbols += [coll.concat, coll.element, coll.unit]
226            symbols += self.constructors.get(sort, ())
227
228            pending.extend(sort for symbol in symbols for sort in self._symbol_sorts(symbol))
229            res.update(symbols)
230        return res

232    def _rewrite_symbols(self) -> set[str]:
233        """Return the set of symbols reffered to in rewrite rules."""
234        res = set()
235
236        # Symbols appearing in rewrite rules are relevant
237        pending = {
238            symbol for rewrite_rule in self.rewrites for symbol in collect_symbols(rewrite_rule.to_axiom().pattern)
239        }
240        while pending:
241            symbol = pending.pop()
242            if symbol in res:
243                continue
244
245            res.add(symbol)
246
247            # Symbols appearing in function rules of a releveant symbol are relevant
248            pending.update(
249                symbol
250                for function_rule in self.functions.get(symbol, ())
251                for symbol in collect_symbols(function_rule.to_axiom().pattern)
252            )
253
254            # If the symbol consumes or produces a collection, collection function symbols are relevant
255            for sort in self._symbol_sorts(symbol):
256                if sort in self.collections:
257                    coll = self.collections[sort]
258                    pending.update([coll.concat, coll.element, coll.unit])
259
260        return res
