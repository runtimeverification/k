161    def __run_proof_hint_process(
162        self,
163        args: str | Iterable[str],
164        *,
165        check: bool = True,
166        input: str | None = None,
167        pipe_stdout: bool = True,
168        pipe_stderr: bool = False,
169        cwd: str | Path | None = None,
170        logger: Logger | None = None,
171        exec_process: bool = False,
172    ) -> CompletedProcess:
173
174        if cwd is not None:
175            cwd = Path(cwd)
176            check_dir_path(cwd)
177
178        if type(args) is str:
179            command = args
180        else:
181            args = tuple(args)
182            command = join(args)
183
184        if not logger:
185            logger = _LOGGER
186
187        proc_stdout = PIPE if pipe_stdout else None
188        proc_stderr = PIPE if pipe_stderr else None
189
190        logger.info(f'Running: {command}')
191
192        if exec_process:
193            stdout.flush()
194            stderr.flush()
195            if type(args) is str:
196                args = split(args)
197            argslist = list(args)
198            execvp(argslist[0], argslist)
199
200        start_time = time()
201
202        res = run(args, input=input, cwd=cwd, stdout=proc_stdout, stderr=proc_stderr, text=False)
203
204        delta_time = time() - start_time
205        logger.info(f'Completed in {delta_time:.3f}s with status {res.returncode}: {command}')
206
207        if check:
208            res.check_returncode()
209
210        return res

212    def run(
213        self,
214        pgm: Pattern,
215        *,
216        cmap: Mapping[str, str] | None = None,
217        pmap: Mapping[str, str] | None = None,
218        term: bool = False,
219        depth: int | None = None,
220        expand_macros: bool = True,
221        search_final: bool = False,
222        no_pattern: bool = False,
223        output: KRunOutput | None = KRunOutput.PRETTY,
224        check: bool = False,
225        pipe_stderr: bool = True,
226        bug_report: BugReport | None = None,
227        save_temps: bool = False,
228        statistics: bool = False,
229        debugger: bool = False,
230    ) -> None:
231        result = self.run_process(
232            pgm,
233            cmap=cmap,
234            pmap=pmap,
235            term=term,
236            depth=depth,
237            expand_macros=expand_macros,
238            search_final=search_final,
239            no_pattern=no_pattern,
240            output=output,
241            pipe_stderr=pipe_stderr,
242            bug_report=bug_report,
243            save_temps=save_temps,
244            statistics=statistics,
245            debugger=debugger,
246        )
247
248        if output != KRunOutput.NONE:
249            output_kore = KoreParser(result.stdout).pattern()
250            match output:
251                case KRunOutput.JSON:
252                    print(self.kore_to_kast(output_kore).to_json())
253                case KRunOutput.KORE:
254                    print(output_kore.text)
255                case KRunOutput.PRETTY | KRunOutput.PROGRAM | KRunOutput.KAST | KRunOutput.BINARY | KRunOutput.LATEX:
256                    print(kore_print(output_kore, definition_dir=self.definition_dir, output=PrintOutput(output.value)))
257                case KRunOutput.NONE:
258                    raise AssertionError()
259
260        if check:
261            result.check_returncode()

304def _krun(
305    command: str = 'krun',
306    *,
307    input_file: Path | None = None,
308    definition_dir: Path | None = None,
309    output: KRunOutput | None = None,
310    parser: str | None = None,
311    depth: int | None = None,
312    cmap: Mapping[str, str] | None = None,
313    pmap: Mapping[str, str] | None = None,
314    term: bool = False,
315    temp_dir: Path | None = None,
316    save_temps: bool = False,
317    statistics: bool = False,
318    no_expand_macros: bool = False,
319    search_final: bool = False,
320    no_pattern: bool = False,
321    # ---
322    check: bool = True,
323    pipe_stderr: bool = True,
324    logger: Logger | None = None,
325    bug_report: BugReport | None = None,
326    debugger: bool = False,
327) -> CompletedProcess:
328    if input_file:
329        check_file_path(input_file)
330
331    if definition_dir:
332        check_dir_path(definition_dir)
333
334    if depth and depth < 0:
335        raise ValueError(f'Expected non-negative depth, got: {depth}')
336
337    if term and (cmap is not None or pmap is not None):
338        raise ValueError('Cannot supply both term and cmap/pmap')
339
340    args = _build_arg_list(
341        command=command,
342        input_file=input_file,
343        definition_dir=definition_dir,
344        output=output,
345        parser=parser,
346        depth=depth,
347        pmap=pmap,
348        cmap=cmap,
349        term=term,
350        temp_dir=temp_dir,
351        save_temps=save_temps,
352        statistics=statistics,
353        no_expand_macros=no_expand_macros,
354        search_final=search_final,
355        no_pattern=no_pattern,
356        debugger=debugger,
357        proof_hint=False,
358    )
359
360    if bug_report is not None:
361        if input_file is not None:
362            new_input_file = Path(f'krun_inputs/{input_file}')
363            bug_report.add_file(input_file, new_input_file)
364            bug_report.add_command([a if a != str(input_file) else str(new_input_file) for a in args])
365        else:
366            bug_report.add_command(args)
367
368    return run_process(args, check=check, pipe_stderr=pipe_stderr, logger=logger or _LOGGER, exec_process=debugger)

371def _build_arg_list(
372    *,
373    command: str,
374    input_file: Path | None,
375    definition_dir: Path | None,
376    output: KRunOutput | None,
377    parser: str | None,
378    depth: int | None,
379    pmap: Mapping[str, str] | None,
380    cmap: Mapping[str, str] | None,
381    term: bool,
382    temp_dir: Path | None,
383    save_temps: bool,
384    statistics: bool,
385    no_expand_macros: bool,
386    search_final: bool,
387    no_pattern: bool,
388    debugger: bool,
389    proof_hint: bool,
390) -> list[str]:
391    args = [command]
392    if input_file:
393        args += [str(input_file)]
394    if definition_dir:
395        args += ['--definition', str(definition_dir)]
396    if output:
397        args += ['--output', output.value]
398    if parser:
399        args += ['--parser', parser]
400    if depth is not None:
401        args += ['--depth', str(depth)]
402    for name, value in (pmap or {}).items():
403        args += [f'-p{name}={value}']
404    for name, value in (cmap or {}).items():
405        args += [f'-c{name}={value}']
406    if term:
407        args += ['--term']
408    if temp_dir:
409        args += ['--temp-dir', str(temp_dir)]
410    if save_temps:
411        args += ['--save-temps']
412    if statistics:
413        args += ['--statistics']
414    if no_expand_macros:
415        args += ['--no-expand-macros']
416    if search_final:
417        args += ['--search-final']
418    if no_pattern:
419        args += ['--no-pattern']
420    if debugger:
421        args += ['--debugger']
422    if proof_hint:
423        args += ['--proof-hint']
424    return args

427def llvm_interpret(
428    definition_dir: str | Path, pattern: Pattern, *, depth: int | None = None, check: bool = True
429) -> Pattern:
430    """Execute the `interpreter` binary generated by the LLVM Backend.
431
432    Args:
433        definition_dir: Path to the kompiled definition directory.
434        pattern: KORE pattern to start rewriting from.
435        depth: Maximal number of rewrite steps to take.
436
437    Returns:
438        The pattern resulting from the rewrites.
439
440    Raises:
441        RuntimeError: If ``check`` and the interpreter fails.
442    """
443    try:
444        res = llvm_interpret_raw(definition_dir, pattern.text, depth=depth, check=check)
445    except CalledProcessError as err:
446        raise RuntimeError(f'Interpreter failed with status {err.returncode}: {err.stderr}') from err
447
448    if not res.stdout:
449        raise RuntimeError(f'Interpreter terminated with status {res.returncode} and produced no output: {res.stderr}')
450
451    return KoreParser(res.stdout).pattern()

454def llvm_interpret_raw(
455    definition_dir: str | Path, kore: str, *, depth: int | None = None, check: bool = True
456) -> CompletedProcess:
457    """Execute the `interpreter` binary generated by the LLVM Backend, with no processing of input/output.
458
459    Args:
460        definition_dir: Path to the kompiled definition directory.
461        pattern: KORE string to start rewriting from.
462        depth: Maximal number of rewrite steps to take.
463        check: check the return code of the CompletedProcess
464
465    Returns:
466        The CompletedProcess of the interpreter.
467
468    Raises:
469        CalledProcessError: If ``check`` and the interpreter fails.
470    """
471    definition_dir = Path(definition_dir)
472    interpreter_file = definition_dir / 'interpreter'
473    check_file_path(interpreter_file)
474
475    depth = depth if depth is not None else -1
476    args = [str(interpreter_file), '/dev/stdin', str(depth), '/dev/stdout']
477
478    return run_process_2(args, input=kore, logger=_LOGGER, loglevel=logging.DEBUG, check=check)
