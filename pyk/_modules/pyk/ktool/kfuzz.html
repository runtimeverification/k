112def kintegers(
113    *,
114    min_value: int | None = None,
115    max_value: int | None = None,
116    with_inj: str | None = None,
117) -> SearchStrategy[Pattern]:
118    """Return a search strategy for K integers.
119
120    Args:
121        min_value: Minimum value for the generated integers
122        max_value: Maximum value for the generated integers
123        with_inj: Return the integer as an injection into this sort
124
125    Returns:
126        A strategy which generates integer domain values.
127    """
128
129    def int_dv(value: int) -> Pattern:
130        res: Pattern = DV(SortApp('SortInt'), value=String(str(value)))
131        if with_inj is not None:
132            res = inj(SortApp('SortInt'), SortApp(f'Sort{with_inj}'), res)
133        return res
134
135    return integers(min_value=min_value, max_value=max_value).map(int_dv)

138def fuzz(
139    definition_dir: str | Path,
140    template: Pattern,
141    subst_strategy: dict[EVar, SearchStrategy[Pattern]],
142    *,
143    check_func: Callable[[Pattern], Any] | None = None,
144    check_exit_code: bool = False,
145    handler: KFuzzHandler = _DEFAULT_HANDLER,
146    subst_func: Callable[[Pattern, Mapping[EVar, Pattern]], Pattern] = substitute_vars,
147    **hypothesis_args: Any,
148) -> None:
149    """Fuzz a property test with concrete execution over a K term.
150
151    Args:
152        definition_dir: The location of the K definition to run the interpreter for.
153        template: The term which will be sent to the interpreter after randomizing inputs. It should contain at least one variable which will be substituted for a value.
154        subst_strategy: Should have each variable in the template term mapped to a strategy for generating values for it.
155        check_func: Will be called on the kore output from the interpreter.
156          Should throw an AssertionError if it determines that the output indicates a test failure.
157          A RuntimeError will be thrown if this is passed as an argument and check_exit_code is True.
158        check_exit_code: Check the exit code of the interpreter for a test failure instead of using check_func.
159          An exit code of 0 indicates a passing test.
160          A RuntimeError will be thrown if this is True and check_func is also passed as an argument.
161        handler: An instance of a `KFuzzHandler` implementing custom behavior while fuzzing.
162        subst_func: A function used to apply the substitution to the template.
163          Takes the original template and a substitution map, and returns the substituted pattern.
164          Defaults to a bottom-up traversal that replaces variables with the corresponding values in the substitution map.
165        hypothesis_args: Keyword arguments that will be passed as settings for the hypothesis test. Defaults:
166
167          deadline: 5000
168
169          phases: (Phase.explicit, Phase.reuse, Phase.generate)
170
171
172    Raises:
173        RuntimeError: If check_func exists and check_exit_code is set, or check_func doesn't exist and check_exit_code is cleared.
174    """
175    if bool(check_func) == check_exit_code:
176        raise RuntimeError('Must pass one of check_func or check_exit_code, and not both!')
177
178    def test(subst_case: Mapping[EVar, Pattern]) -> None:
179        def sub(p: Pattern) -> Pattern:
180            if isinstance(p, EVar) and p in subst_case:
181                return subst_case[p]
182            else:
183                return p
184
185        handler.handle_test(subst_case)
186        test_pattern = subst_func(template, subst_case)
187        res = llvm_interpret_raw(definition_dir, test_pattern.text, check=False)
188
189        try:
190            if check_exit_code:
191                assert res.returncode == 0
192            else:
193                assert check_func
194                res_pattern = KoreParser(res.stdout).pattern()
195                check_func(res_pattern)
196        except AssertionError:
197            handler.handle_failure(subst_case)
198            raise
199
200    strat: SearchStrategy = fixed_dictionaries(subst_strategy)
201
202    # Default settings for hypothesis
203    hypothesis_args.setdefault('deadline', 5000)
204    hypothesis_args.setdefault('phases', (Phase.explicit, Phase.reuse, Phase.generate))
205
206    given(strat)(settings(**hypothesis_args)(test))()
