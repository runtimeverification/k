 49def _kprove(
 50    spec_file: Path,
 51    *,
 52    command: Iterable[str] = ('kprove',),
 53    kompiled_dir: Path | None = None,
 54    spec_module_name: str | None = None,
 55    md_selector: str | None = None,
 56    include_dirs: Iterable[Path] = (),
 57    emit_json_spec: Path | None = None,
 58    output: KProveOutput | None = None,
 59    depth: int | None = None,
 60    claims: Iterable[str] = (),
 61    type_inference_mode: str | TypeInferenceMode | None = None,
 62    temp_dir: Path | None = None,
 63    haskell_backend_command: str | None = None,
 64    dry_run: bool = False,
 65    # --
 66    args: Iterable[str] = (),
 67    # --
 68    env: Mapping[str, str] | None = None,
 69    check: bool = True,
 70) -> CompletedProcess:
 71    check_file_path(spec_file)
 72
 73    for include_dir in include_dirs:
 74        check_dir_path(include_dir)
 75
 76    if depth is not None and depth < 0:
 77        raise ValueError(f'Argument "depth" must be non-negative, got: {depth}')
 78
 79    if type_inference_mode is not None:
 80        type_inference_mode = TypeInferenceMode(type_inference_mode)
 81
 82    typed_args = _build_arg_list(
 83        kompiled_dir=kompiled_dir,
 84        spec_module_name=spec_module_name,
 85        md_selector=md_selector,
 86        include_dirs=include_dirs,
 87        emit_json_spec=emit_json_spec,
 88        output=output,
 89        depth=depth,
 90        claims=claims,
 91        type_inference_mode=type_inference_mode,
 92        temp_dir=temp_dir,
 93        haskell_backend_command=haskell_backend_command,
 94        dry_run=dry_run,
 95    )
 96
 97    run_args = tuple(chain(command, [str(spec_file)], typed_args, args))
 98
 99    return run_process_2(run_args, logger=_LOGGER, env=env, check=check)

102def _build_arg_list(
103    *,
104    kompiled_dir: Path | None,
105    spec_module_name: str | None,
106    md_selector: str | None,
107    include_dirs: Iterable[Path],
108    emit_json_spec: Path | None,
109    output: KProveOutput | None,
110    depth: int | None,
111    claims: Iterable[str],
112    type_inference_mode: TypeInferenceMode | None,
113    temp_dir: Path | None,
114    haskell_backend_command: str | None,
115    dry_run: bool,
116) -> list[str]:
117    args = []
118
119    if kompiled_dir:
120        args += ['--definition', str(kompiled_dir)]
121
122    if spec_module_name:
123        args += ['--spec-module', spec_module_name]
124
125    if md_selector:
126        args += ['--md-selector', md_selector]
127
128    for include_dir in include_dirs:
129        args += ['-I', str(include_dir)]
130
131    if emit_json_spec:
132        args += ['--emit-json-spec', str(emit_json_spec)]
133
134    if output:
135        args += ['--output', output.value]
136
137    if claims:
138        args += ['--claims', ','.join(claims)]
139
140    if type_inference_mode:
141        args += ['--type-inference-mode', type_inference_mode.value]
142
143    if temp_dir:
144        args += ['--temp-dir', str(temp_dir)]
145
146    if haskell_backend_command:
147        args += ['--haskell-backend-command', haskell_backend_command]
148
149    if depth:
150        args += ['--depth', str(depth)]
151
152    if dry_run:
153        args.append('--dry-run')
154
155    return args

185    def prove(
186        self,
187        spec_file: Path,
188        spec_module_name: str | None = None,
189        args: Iterable[str] = (),
190        include_dirs: Iterable[Path] = (),
191        md_selector: str | None = None,
192        haskell_args: Iterable[str] = (),
193        depth: int | None = None,
194    ) -> list[CTerm]:
195        log_file = spec_file.with_suffix('.debug-log')
196        if log_file.exists():
197            log_file.unlink()
198        haskell_log_args = [
199            '--log',
200            str(log_file),
201            '--log-format',
202            KoreExecLogFormat.ONELINE.value,
203            '--log-entries',
204            'DebugTransition',
205        ]
206
207        env = os.environ.copy()
208        existing_opts = os.getenv('KORE_EXEC_OPTS')
209        kore_exec_opts = ' '.join(list(haskell_args) + haskell_log_args + ([existing_opts] if existing_opts else []))
210        _LOGGER.debug(f'export KORE_EXEC_OPTS={kore_exec_opts!r}')
211        env['KORE_EXEC_OPTS'] = kore_exec_opts
212
213        proc_result = _kprove(
214            spec_file=spec_file,
215            command=self.prover,
216            kompiled_dir=self.definition_dir,
217            spec_module_name=spec_module_name,
218            include_dirs=include_dirs,
219            md_selector=md_selector,
220            output=KProveOutput.JSON,
221            temp_dir=self.use_directory,
222            args=self.prover_args + list(args),
223            env=env,
224            check=False,
225            depth=depth,
226        )
227
228        if proc_result.returncode not in (0, 1):
229            raise RuntimeError('kprove failed!')
230
231        debug_log = _get_rule_log(log_file)
232        final_state = KInner.from_dict(kast_term(json.loads(proc_result.stdout)))
233        if is_top(final_state) and len(debug_log) == 0:
234            raise ValueError(f'Proof took zero steps, likely the LHS is invalid: {spec_file}')
235        return [CTerm.from_kast(disjunct) for disjunct in flatten_label('#Or', final_state)]

324    @contextmanager
325    def _tmp_claim_definition(
326        self,
327        claim: KClaim,
328        claim_id: str,
329        lemmas: Iterable[KRule] = (),
330    ) -> Iterator[tuple[Path, str]]:
331        with self._temp_file(suffix='-spec.k') as ntf:
332            tmp_claim_file = Path(ntf.name)
333            tmp_module_name = tmp_claim_file.stem.removesuffix('-spec').rstrip('_').replace('_', '-').upper() + '-SPEC'
334            tmp_module_name = re.sub(r'-+', '-', tmp_module_name)
335
336            sentences: list[KRuleLike] = []
337            sentences += lemmas
338            sentences += [claim]
339
340            claim_module = KFlatModule(tmp_module_name, sentences, imports=[KImport(self.main_module, True)])
341            requires = [KRequire(str(self.main_file))] if self.main_file is not None else []
342            claim_definition = KDefinition(tmp_module_name, [claim_module], requires=requires)
343
344            ntf.write(gen_file_timestamp() + '\n')
345            ntf.write(self.pretty_print(claim_definition) + '\n\n')
346            ntf.flush()
347
348            _LOGGER.info(f'Wrote claim file: {tmp_claim_file}')
349            yield tmp_claim_file, tmp_module_name

352def _get_rule_log(debug_log_file: Path) -> list[list[tuple[str, bool, int]]]:
353    # rule_loc, is_success, ellapsed_time_since_start
354    def _get_rule_line(_line: str) -> tuple[str, bool, int] | None:
355        if _line.startswith('kore-exec: ['):
356            time = int(_line.split('[')[1].split(']')[0])
357            if _line.find('(DebugTransition): after  apply axioms: ') > 0:
358                rule_name = ':'.join(_line.split(':')[-4:]).strip()
359                return (rule_name, True, time)
360            elif _line.find('(DebugAttemptedRewriteRules): ') > 0:
361                rule_name = ':'.join(_line.split(':')[-4:]).strip()
362                return (rule_name, False, time)
363        return None
364
365    log_lines: list[tuple[str, bool, int]] = []
366    with open(debug_log_file) as log_file:
367        for line in log_file.read().split('\n'):
368            if processed_line := _get_rule_line(line):
369                log_lines.append(processed_line)
370
371    # rule_loc, is_success, time_delta
372    axioms: list[list[tuple[str, bool, int]]] = [[]]
373    just_applied = True
374    prev_time = 0
375    for rule_name, is_application, rule_time in log_lines:
376        rtime = rule_time - prev_time
377        prev_time = rule_time
378        if not is_application:
379            if just_applied:
380                axioms.append([])
381            just_applied = False
382        else:
383            just_applied = True
384        axioms[-1].append((rule_name, is_application, rtime))
385
386    if len(axioms[-1]) == 0:
387        axioms.pop(-1)
388
389    return axioms
