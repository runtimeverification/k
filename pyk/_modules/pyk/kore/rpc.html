 110    @staticmethod
 111    def _create_connection(host: str, port: int, timeout: int | None) -> socket.socket:
 112        if timeout is not None and timeout < 0:
 113            raise ValueError(f'Expected nonnegative timeout value, got: {timeout}')
 114
 115        _LOGGER.info(f'Connecting to host: {host}:{port}')
 116
 117        timeout_datetime = datetime.now() + timedelta(milliseconds=timeout) if timeout is not None else None
 118        while timeout_datetime is None or datetime.now() < timeout_datetime:
 119            try:
 120                sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 121                sock.connect((host, port))
 122                _LOGGER.info(f'Connected to host: {host}:{port}')
 123                return sock
 124            except ConnectionRefusedError:
 125                sock.close()
 126                sleep(0.1)
 127
 128        raise RuntimeError(f'Connection timed out: {host}:{port}')

 184    def __init__(
 185        self,
 186        default_host: str,
 187        default_port: int,
 188        default_transport: TransportType,
 189        dispatch: dict[str, list[tuple[str, int, TransportType]]],
 190        *,
 191        timeout: int | None = None,
 192        bug_report: BugReport | None = None,
 193        bug_report_id: str | None = None,
 194    ):
 195        client_cache = {}
 196        self._clients = {}
 197        self._default_client = JsonRpcClient(
 198            default_host,

 201            bug_report=bug_report,
 202            bug_report_id=bug_report_id,
 203            transport=default_transport,
 204        )
 205        client_cache[(default_host, default_port)] = self._default_client
 206        for method, servers in dispatch.items():
 207            for host, port, transport in servers:
 208                if (host, port) in client_cache:
 209                    self._update_clients(method, client_cache[(host, port)])
 210                else:
 211                    new_id = None if bug_report_id is None else bug_report_id + '_' + str(transport)
 212                    new_client = JsonRpcClient(
 213                        host, port, timeout=timeout, bug_report=bug_report, bug_report_id=new_id, transport=transport
 214                    )
 215                    self._update_clients(method, new_client)
 216                    client_cache[(host, port)] = new_client

 292    def request(self, method: str, **params: Any) -> dict[str, Any]:
 293        req_id = f'{id(self)}-{self._req_id:03}'
 294        self._req_id += 1
 295
 296        payload = {
 297            'jsonrpc': self._JSON_RPC_VERSION,
 298            'id': req_id,
 299            'method': method,
 300            'params': params,
 301        }
 302
 303        req = json.dumps(payload)
 304
 305        base_name = self._bug_report_id if self._bug_report_id is not None else 'kore_rpc'
 306        req_name = f'{base_name}/{id(self)}/{req_id}'
 307        if self._bug_report:
 308            bug_report_request = f'{req_name}_request.json'
 309            self._bug_report.add_file_contents(req, Path(bug_report_request))
 310            self._bug_report.add_request(f'{req_name}_request.json')
 311
 312        resp = self._transport.request(req, req_id, method)
 313        if not resp:
 314            raise RuntimeError('Empty response received')
 315
 316        if self._bug_report:
 317            bug_report_response = f'{req_name}_response.json'
 318            self._bug_report.add_file_contents(resp, Path(bug_report_response))
 319            self._bug_report.add_request(f'{req_name}_response.json')
 320
 321        data = json.loads(resp)
 322        self._check(data)
 323        assert data['id'] == req_id
 324
 325        return data['result']


 326
 327    @staticmethod
 328    def _check(response: Mapping[str, Any]) -> None:
 329        if 'error' not in response:
 330            return
 331
 332        assert response['error']['code'] not in {-32700, -32600}, 'Malformed JSON-RPC request'
 333        raise JsonRpcError(**response['error'])

 490    @staticmethod
 491    def _subst_to_dict(pattern: Pattern) -> dict[EVar, Pattern]:
 492        def extract_entry(pattern: Pattern) -> tuple[EVar, Pattern]:
 493            if not isinstance(pattern, Equals):
 494                raise ValueError(fr'Expected \equals as substituion entry, got: {pattern.text}')
 495            if pattern.sort != SORT_GENERATED_TOP_CELL:
 496                raise ValueError(
 497                    f'Expected {SORT_GENERATED_TOP_CELL.text} as substitution entry sort, got: {pattern.sort.text}'
 498                )
 499            if not isinstance(pattern.left, EVar):
 500                raise ValueError(f'Expected EVar as substitution entry key, got: {pattern.left.text}')
 501            if pattern.left.sort != pattern.op_sort:
 502                raise ValueError(
 503                    f'Mismatch between substitution entry and key sort: {pattern.op_sort.text} and {pattern.left.sort.text}'
 504                )
 505            return pattern.left, pattern.right
 506
 507        res: dict[EVar, Pattern] = {}
 508        for conjunct in manip.conjuncts(pattern):
 509            key, value = extract_entry(conjunct)
 510            if key in res:
 511                raise ValueError(f'Duplicate substitution entry key: {key.text} -> {[res[key].text, value.text]}')
 512            res[key] = value
 513        return res
 514
 515    @staticmethod
 516    def _dict_to_subst(dct: Mapping[EVar, Pattern]) -> And:
 517        return And(
 518            SORT_GENERATED_TOP_CELL,
 519            tuple(Equals(var.sort, SORT_GENERATED_TOP_CELL, var, val) for var, val in dct.items()),
 520        )
 521
 522    @property
 523    def kore(self) -> Pattern:
 524        _kore = self.term
 525        if self.substitution is not None:
 526            _kore = And(SORT_GENERATED_TOP_CELL, (_kore,) + self._dict_to_subst(self.substitution).ops)
 527        if self.predicate is not None:
 528            _kore = And(SORT_GENERATED_TOP_CELL, (_kore, self.predicate))
 529        return _kore

 928        assert err.code not in {-32601, -32602}, 'Malformed Kore-RPC request'
 929        match err.code:
 930            case 1:

 932            case 2:

 934            case 3:

 938            case 5:
