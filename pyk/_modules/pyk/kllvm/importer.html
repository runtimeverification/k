20def rtld_local() -> Iterator[None]:

23    yield

27def import_from_file(module_name: str, module_file: str | Path) -> ModuleType:
28    module_file = Path(module_file).resolve()
29    check_file_path(module_file)
30
31    spec = spec_from_file_location(module_name, module_file)
32    if not spec:
33        raise ValueError('Could not create ModuleSpec')
34
35    module = module_from_spec(spec)
36    if not module:
37        raise ValueError('Could not create ModuleType')
38
39    if not spec.loader:
40        raise ValueError('Spec has no loader')
41
42    spec.loader.exec_module(module)
43
44    return module
