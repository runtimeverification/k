220    def _info_text(self) -> str:
221        term_str = '✅' if self._term_on else '❌'
222        constraint_str = '✅' if self._constraint_on else '❌'
223        custom_str = '' if self._custom_view is None else f'{"✅" if self._custom_on else "❌"} Custom View.'
224        minimize_str = '✅' if self._minimize else '❌'
225        status_str = '✅' if self._status_on else '❌'
226        element_str = 'NOTHING'
227        if type(self._element) is KCFG.Node:
228            element_str = f'node({shorten_hashes(self._element.id)})'
229        elif type(self._element) is KCFG.Edge:
230            element_str = f'edge({shorten_hashes(self._element.source.id)},{shorten_hashes(self._element.target.id)})'
231        elif type(self._element) is KCFG.MergedEdge:
232            element_str = (
233                f'merged_edge({shorten_hashes(self._element.source.id)},{shorten_hashes(self._element.target.id)})'
234            )
235        elif type(self._element) is KCFG.Cover:
236            element_str = f'cover({shorten_hashes(self._element.source.id)},{shorten_hashes(self._element.target.id)})'
237        return f'{element_str} selected. {minimize_str} Minimize Output. {term_str} Term View. {constraint_str} Constraint View. {status_str} Status View. {custom_str}'

251    def toggle_option(self, field: str) -> bool:
252        assert field in ['minimize', 'term_on', 'constraint_on', 'custom_on', 'status_on']
253        field_attr = f'_{field}'
254        old_value = getattr(self, field_attr)
255        new_value = not old_value
256        # Do not turn on custom view if it's not available
257        if field == 'custom_on' and self._custom_view is None:
258            new_value = False
259        setattr(self, field_attr, new_value)
260        if field == 'minimize':
261            self._cterm_show = self._cterm_show.let(minimize=self._minimize)
262        self._update()
263        return new_value

279    def _update(self) -> None:
280        def _boolify(c: KInner) -> KInner:
281            if type(c) is KApply and c.label.name == '#Equals' and c.args[0] == TRUE:
282                return c.args[1]
283            else:
284                return c
285
286        def _boolify_print(c: KInner) -> str:
287            return PrettyPrinter(self._definition).print(_boolify(c))
288
289        def _cterm_text(cterm: CTerm) -> tuple[str, str]:
290            config_text = self._cterm_show.show_config(cterm)
291            constraint_text = self._cterm_show.show_constraints(cterm)
292            return '\n'.join(config_text), '\n'.join(constraint_text)
293
294        def _csubst_text(csubst: CSubst) -> tuple[str, str]:
295            equalities = map(
296                _boolify_print, flatten_label('#And', csubst.pred(sort_with=self._definition, constraints=False))
297            )
298            constraints = map(_boolify_print, flatten_label('#And', csubst.constraint))
299            return '\n'.join(equalities), '\n'.join(constraints)
300
301        term_str = 'Term'
302        constraint_str = 'Constraint'
303        custom_str = 'Custom'
304
305        if self._element is not None:
306            if type(self._element) is KCFG.Node:
307                term_str, constraint_str = _cterm_text(self._element.cterm)
308
309            elif type(self._element) is KCFG.Edge:
310                config_source, *constraints_source = self._element.source.cterm
311                config_target, *constraints_target = self._element.target.cterm
312                constraints_new = [c for c in constraints_target if c not in constraints_source]
313                config = push_down_rewrites(KRewrite(config_source, config_target))
314                crewrite = CTerm(config, constraints_new)
315                term_str, constraint_str = _cterm_text(crewrite)
316
317            elif type(self._element) is KCFG.MergedEdge:
318                config_source, *constraints_source = self._element.source.cterm
319                config_target, *constraints_target = self._element.target.cterm
320                constraints_new = [c for c in constraints_target if c not in constraints_source]
321                config = push_down_rewrites(KRewrite(config_source, config_target))
322                crewrite = CTerm(config, constraints_new)
323                term_str, constraint_str = _cterm_text(crewrite)
324
325            elif type(self._element) is KCFG.Cover:
326                term_str, constraint_str = _csubst_text(self._element.csubst)
327
328            elif type(self._element) is KCFG.Split:
329                term_strs = [f'split: {shorten_hashes(self._element.source.id)}']
330                for target_id, csubst in self._element.splits.items():
331                    term_strs.append('')
332                    term_strs.append(f'  - {shorten_hashes(target_id)}')
333                    equalities_str, constraints_str = _csubst_text(csubst)
334                    if equalities_str != '#Top':
335                        term_strs.extend([f'    {es}' for es in equalities_str.split('\n')])
336                    if constraints_str != '#Top':
337                        term_strs.extend([f'    {es}' for es in constraints_str.split('\n')])
338                term_str = '\n'.join(term_strs)
339
340            elif type(self._element) is KCFG.NDBranch:
341                term_strs = [f'ndbranch: {shorten_hashes(self._element.source.id)}']
342                for target in self._element.targets:
343                    term_strs.append('')
344                    term_strs.append(f'  - {shorten_hashes(target.id)}')
345                    term_strs.append('    (1 step)')
346                term_str = '\n'.join(term_strs)
347
348            if self._custom_view is not None:
349                # To appease the type-checker
350                if type(self._element) is KCFG.Node:
351                    custom_str = '\n'.join(self._custom_view(self._element))
352                elif isinstance(self._element, KCFG.Successor):
353                    custom_str = '\n'.join(self._custom_view(self._element))
354
355        self.query_one('#info', Info).text = self._info_text()
356        self.query_one('#term', Term).text = term_str
357        self.query_one('#constraint', Constraint).text = constraint_str
358        if self._custom_view is not None:
359            self.query_one('#custom', Custom).text = custom_str
360        self.query_one('#status', Status).text = self._status_text()

492    def action_keystroke(self, key: str) -> None:
493        if key == 'h':
494            if self._selected_chunk is not None and self._selected_chunk.startswith('node_'):
495                node_id = self._selected_chunk[5:]
496                self._hidden_chunks.append(self._selected_chunk)
497                self.query_one(f'#{self._selected_chunk}', GraphChunk).add_class('hidden')
498                self.query_one('#info', Info).text = f'HIDDEN: node({shorten_hashes(node_id)})'
499        elif key == 'H':
500            for hc in self._hidden_chunks:
501                self.query_one(f'#{hc}', GraphChunk).remove_class('hidden')
502            node_ids = [nid[5:] for nid in self._hidden_chunks]
503            self.query_one('#info', Info).text = f'UNHIDDEN: nodes({shorten_hashes(node_ids)})'
504            self._hidden_chunks = []
505        elif key in ['term', 'constraint', 'status']:
506            self.query_one('#node-view', NodeView).toggle_view(key)
507        elif key == 'custom' and self._custom_view is not None:
508            self.query_one('#node-view', NodeView).toggle_view(key)
509        elif key in ['minimize']:
510            self.query_one('#node-view', NodeView).toggle_option(key)
