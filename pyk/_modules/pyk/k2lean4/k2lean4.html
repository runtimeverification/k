 58    def _sort_block(self, sorts: list[str]) -> Command | None:
 59        """Return an optional mutual block or declaration."""
 60        commands: tuple[Command, ...] = tuple(
 61            self._transform_sort(sort) for sort in sorts if sort not in _PRELUDE_SORTS
 62        )
 63
 64        if not commands:
 65            return None
 66
 67        if len(commands) == 1:
 68            (command,) = commands
 69            return command
 70
 71        return Mutual(commands=commands)

 73    def _transform_sort(self, sort: str) -> Declaration:
 74        def is_inductive(sort: str) -> bool:
 75            decl = self.defn.sorts[sort]
 76            return not decl.hooked and 'hasDomainValues' not in decl.attrs_by_key and not self._is_cell(sort)
 77
 78        def is_collection(sort: str) -> bool:
 79            return sort in self.defn.collections
 80
 81        if is_inductive(sort):
 82            return self._inductive(sort)
 83
 84        if self._is_cell(sort):
 85            return self._cell(sort)
 86
 87        if is_collection(sort):
 88            return self._collection(sort)
 89
 90        raise AssertionError

 92    def _inductive(self, sort: str) -> Inductive:
 93        subsorts = sorted(self.defn.subsorts.get(sort, ()))
 94        symbols = sorted(self.defn.constructors.get(sort, ()))
 95        ctors: list[Ctor] = []
 96        ctors.extend(self._inj_ctor(sort, subsort) for subsort in subsorts)
 97        ctors.extend(self._symbol_ctor(sort, symbol) for symbol in symbols)
 98        return Inductive(sort, Signature((), Term('Type')), ctors=ctors)

123    def _cell(self, sort: str) -> Structure:
124        (cell_ctor,) = self.defn.constructors[sort]
125        decl = self.defn.symbols[cell_ctor]
126        param_sorts = _param_sorts(decl)
127
128        param_names: list[str]
129
130        if all(self._is_cell(sort) for sort in param_sorts):
131            param_names = []
132            for param_sort in param_sorts:
133                assert param_sort.startswith('Sort')
134                assert param_sort.endswith('Cell')
135                name = param_sort[4:-4]
136                name = name[0].lower() + name[1:]
137                param_names.append(name)
138        else:
139            assert len(param_sorts) == 1
140            param_names = ['val']
141
142        fields = tuple(ExplBinder((name,), Term(sort)) for name, sort in zip(param_names, param_sorts, strict=True))
143        return Structure(sort, Signature((), Term('Type')), ctor=StructCtor(fields))

145    def _collection(self, sort: str) -> Structure:
146        coll = self.defn.collections[sort]
147        elem = self.defn.symbols[coll.element]
148        sorts = _param_sorts(elem)
149        val: Term
150        match coll.kind:
151            case CollectionKind.LIST:
152                (item,) = sorts
153                val = Term(f'List {item}')
154            case CollectionKind.SET:
155                (item,) = sorts
156                val = Term(f'(SetHook {item}).set')
157            case CollectionKind.MAP:
158                key, value = sorts
159                val = Term(f'(MapHook {key} {value}).map')
160        field = ExplBinder(('coll',), val)
161        return Structure(sort, Signature((), Term('Type')), ctor=StructCtor((field,)))

185    def _inj_val(self, subsort: str, supersort: str) -> FieldVal:
186        subsubsorts: list[str]
187        if subsort.endswith('CellMap'):
188            subsubsorts = []  # Disregard injection from value sort to cell map sort
189        else:
190            subsubsorts = sorted(self.defn.subsorts.get(subsort, []))

197    def _inj_alts(self, subsort: str, supersort: str, subsubsorts: list[str]) -> list[Alt]:
198        def inj(subsort: str, supersort: str, x: str) -> Term:
199            return Term(f'{supersort}.inj_{subsort} {x}')
200
201        res = []
202        for subsubsort in subsubsorts:
203            res.append(Alt((inj(subsubsort, subsort, 'x'),), inj(subsubsort, supersort, 'x')))

205        if self.defn.constructors.get(subsort, []):
206            # Has actual constructors, not only subsorts
207            default = Alt((Term('x'),), inj(subsort, supersort, 'x'))
208            res.append(default)

216    def _transform_func(self, func: str) -> Axiom:
217        ident = self._symbol_ident(func)
218        decl = self.defn.symbols[func]
219        sort_params = [var.name for var in decl.symbol.vars]
220        param_sorts = [sort.name for sort in decl.param_sorts]
221        sort = decl.sort.name
222
223        binders: list[Binder] = []
224        if sort_params:
225            binders.append(ImplBinder(sort_params, Term('Type')))
226        binders.extend(ExplBinder((f'x{i}',), Term(sort)) for i, sort in enumerate(param_sorts))
227        return Axiom(ident, Signature(binders, Term(f'Option {sort}')))

236def _ordered_sorts(defn: KoreDefn) -> list[list[str]]:
237    deps = _sort_dependencies(defn)
238    sccs = _sort_sccs(deps)
239
240    sorts_by_scc: dict[int, set[str]] = {}
241    for sort, scc in sccs.items():
242        sorts_by_scc.setdefault(scc, set()).add(sort)
243
244    scc_deps: dict[int, set[int]] = {}
245    for scc, sorts in sorts_by_scc.items():
246        assert sorts
247        sort, *_ = sorts
248        scc_deps[scc] = set()
249        for dep in deps[sort]:
250            dep_scc = sccs[dep]
251            if dep_scc == scc:
252                continue
253            scc_deps[scc].add(dep_scc)
254
255    ordered_sccs = list(TopologicalSorter(scc_deps).static_order())
256    return [sorted(sorts_by_scc[scc]) for scc in ordered_sccs]

259def _sort_dependencies(defn: KoreDefn) -> dict[str, set[str]]:
260    """Transitively closed sort dependency graph."""
261    sorts = defn.sorts
262    deps: list[tuple[str, str]] = []
263    for sort in sorts:
264        # A sort depends on its subsorts (due to inj_{subsort} constructors)
265        deps.extend((sort, subsort) for subsort in defn.subsorts.get(sort, []))
266        # A sort depends on the parameter sorts of all its constructors
267        deps.extend(
268            (sort, param_sort)
269            for ctor in defn.constructors.get(sort, [])
270            for param_sort in _param_sorts(defn.symbols[ctor])
271        )
272        # If the sort is a collection, the element function parameters are dependencies
273        if sort in defn.collections:
274            coll = defn.collections[sort]
275            elem = defn.symbols[coll.element]
276            deps.extend((sort, param_sort) for param_sort in _param_sorts(elem))
277
278    closed = POSet(deps).image  # TODO POSet should be called "transitively closed relation" or similar
279    res = {
280        sort: set(closed.get(sort, [])) for sort in sorts
281    }  # Ensure that sorts without dependencies are also represented
282    return res

286def _sort_sccs(deps: dict[str, set[str]]) -> dict[str, int]:
287    res: dict[str, int] = {}
288
289    scc = count()
290    for sort, dep_sorts in deps.items():
291        if sort in res:
292            continue
293        idx = next(scc)
294        res[sort] = idx
295        for dep in dep_sorts:
296            if sort in deps[dep]:
297                res[dep] = idx
298
299    return res
