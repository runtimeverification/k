 96    def execute(
 97        self,
 98        cterm: CTerm,
 99        depth: int | None = None,
100        cut_point_rules: Iterable[str] | None = None,
101        terminal_rules: Iterable[str] | None = None,
102        module_name: str | None = None,
103    ) -> CTermExecute:
104
105        _LOGGER.debug(f'Executing: {cterm}')
106        kore = self.kast_to_kore(cterm.kast)
107        try:
108            response = self._kore_client.execute(
109                kore,
110                max_depth=depth,
111                cut_point_rules=cut_point_rules,
112                terminal_rules=terminal_rules,
113                module_name=module_name,
114                log_successful_rewrites=self._log_succ_rewrites,
115                log_failed_rewrites=self._log_succ_rewrites and self._log_fail_rewrites,
116            )
117        except SmtSolverError as err:
118            raise self._smt_solver_error(err) from err
119
120        if isinstance(response, AbortedResult):
121            unknown_predicate = response.unknown_predicate.text if response.unknown_predicate else None
122            raise ValueError(f'Backend responded with aborted state. Unknown predicate: {unknown_predicate}')
123
124        state = CTerm.from_kast(self.kore_to_kast(response.state.kore))
125        resp_next_states = response.next_states or ()
126        branching_constraints = tuple(
127            self.kore_to_kast(not_none(s.rule_predicate)) if s.rule_predicate is not None else None
128            for s in resp_next_states
129        )
130        next_states = tuple(
131            NextState(CTerm.from_kast(self.kore_to_kast(ns.kore)), c)
132            for ns, c in zip(resp_next_states, branching_constraints, strict=True)
133        )
134
135        assert all(not cterm.is_bottom for cterm, _ in next_states)
136        assert len(next_states) != 1 or response.reason is StopReason.CUT_POINT_RULE
137
138        return CTermExecute(
139            state=state,
140            next_states=next_states,
141            depth=response.depth,
142            vacuous=response.reason is StopReason.VACUOUS,
143            logs=response.logs,
144        )

162    def get_model(self, cterm: CTerm, module_name: str | None = None) -> Subst | None:
163        _LOGGER.debug(f'Getting model: {cterm}')
164        kore = self.kast_to_kore(cterm.kast)
165        try:
166            result = self._kore_client.get_model(kore, module_name=module_name)
167        except SmtSolverError as err:
168            raise self._smt_solver_error(err) from err
169
170        if type(result) is UnknownResult:
171            _LOGGER.debug('Result is Unknown')
172            return None
173        elif type(result) is UnsatResult:
174            _LOGGER.debug('Result is UNSAT')
175            return None
176        elif type(result) is SatResult:
177            _LOGGER.debug('Result is SAT')
178            if not result.model:
179                return Subst({})
180            model_subst = self.kore_to_kast(result.model)
181            try:
182                return Subst.from_pred(model_subst)
183            except ValueError as err:
184                raise AssertionError(f'Received a non-substitution from get-model endpoint: {model_subst}') from err
185
186        else:
187            raise AssertionError('Received an invalid response from get-model endpoint')

189    def implies(
190        self,
191        antecedent: CTerm,
192        consequent: CTerm,
193        bind_universally: bool = False,
194        failure_reason: bool = False,
195        module_name: str | None = None,
196        assume_defined: bool = False,
197    ) -> CTermImplies:
198        _LOGGER.debug(f'Checking implication: {antecedent} #Implies {consequent}')
199        _consequent = consequent.kast
200        unbound_consequent = [v for v in consequent.free_vars if v not in antecedent.free_vars]
201        if len(unbound_consequent) > 0:
202            bind_text, bind_label = ('existentially', '#Exists')
203            if bind_universally:
204                bind_text, bind_label = ('universally', '#Forall')
205            _LOGGER.debug(f'Binding variables in consequent {bind_text}: {unbound_consequent}')
206            for uc in unbound_consequent:
207                # Setting Sort1 to KItem in #Exists to avoid inferring the type of each uc.
208                # This should not have any effect on the resulting KORE pattern (\exists only has Sort2 as sort variable).
209                _consequent = KApply(KLabel(bind_label, [K_ITEM, GENERATED_TOP_CELL]), [KVariable(uc), _consequent])
210        antecedent_kore = self.kast_to_kore(antecedent.kast)
211        consequent_kore = self.kast_to_kore(_consequent)
212        try:
213            result = self._kore_client.implies(
214                antecedent_kore, consequent_kore, module_name=module_name, assume_defined=assume_defined
215            )
216        except SmtSolverError as err:
217            raise self._smt_solver_error(err) from err
218
219        if not result.valid:
220            if result.substitution is not None:
221                _LOGGER.debug(f'Received a non-empty substitution for falsifiable implication: {result.substitution}')
222            if result.predicate is not None:
223                _LOGGER.debug(f'Received a non-empty predicate for falsifiable implication: {result.predicate}')
224            failing_cells: list[tuple[str, KInner]] = []
225            remaining_implication: KInner | None = None
226            if failure_reason:
227                _config_match = self.implies(
228                    CTerm.from_kast(antecedent.config),
229                    CTerm.from_kast(consequent.config),
230                    bind_universally=bind_universally,
231                    failure_reason=False,
232                    module_name=module_name,
233                    assume_defined=assume_defined,
234                )
235                config_match = _config_match.csubst
236                if config_match is None:
237                    curr_cell_match = Subst({})
238                    for cell in antecedent.cells:
239                        antecedent_cell = sort_ac_collections(antecedent.cell(cell))
240
241                        if cell not in consequent.cells:
242                            failing_cells.append((cell, KRewrite(antecedent_cell, KToken('.K', sort='KItem'))))
243                        else:
244                            consequent_cell = sort_ac_collections(consequent.cell(cell))
245                            cell_match = consequent_cell.match(antecedent_cell)
246                            if cell_match is not None:
247                                _curr_cell_match = curr_cell_match.union(cell_match)
248                                if _curr_cell_match is not None:
249                                    curr_cell_match = _curr_cell_match
250                                    continue
251                            failing_cells.append((cell, KRewrite(antecedent_cell, consequent_cell)))
252                else:
253                    consequent_constraints = list(
254                        filter(
255                            lambda x: not is_spurious_constraint(x),
256                            map(config_match.subst, consequent.constraints),
257                        )
258                    )
259                    remaining_implication = CTerm._ml_impl(antecedent.constraints, consequent_constraints)
260            return CTermImplies(None, tuple(failing_cells), remaining_implication, result.logs)
261
262        if result.substitution is None:
263            raise ValueError('Received empty substutition for valid implication.')
264        if result.predicate is None:
265            raise ValueError('Received empty predicate for valid implication.')
266        ml_subst = self.kore_to_kast(result.substitution)
267        ml_pred = self.kore_to_kast(result.predicate)
268        ml_subst_pred = mlAnd(flatten_label('#And', ml_subst) + flatten_label('#And', ml_pred))
269        csubst = CSubst.from_pred(ml_subst_pred)
270        return CTermImplies(csubst, (), None, result.logs)

272    def assume_defined(self, cterm: CTerm, module_name: str | None = None) -> CTerm:
273        _LOGGER.debug(f'Computing definedness condition for: {cterm}')
274        cterm_simplified, logs = self.simplify(cterm, module_name=module_name)
275        kast = KApply(KLabel('#Ceil', [GENERATED_TOP_CELL, GENERATED_TOP_CELL]), [cterm_simplified.config])
276        kast_simplified, logs = self.kast_simplify(kast, module_name=module_name)
277        _LOGGER.debug(f'Definedness condition computed: {kast_simplified}')
278        return cterm.add_constraint(kast_simplified)

290@contextmanager
291def cterm_symbolic(
292    definition: KDefinition,
293    definition_dir: Path,
294    *,
295    id: str | None = None,
296    port: int | None = None,
297    kore_rpc_command: str | Iterable[str] | None = None,
298    llvm_definition_dir: Path | None = None,
299    smt_timeout: int | None = None,
300    smt_retry_limit: int | None = None,
301    smt_tactic: str | None = None,
302    bug_report: BugReport | None = None,
303    haskell_log_format: KoreExecLogFormat = KoreExecLogFormat.ONELINE,
304    haskell_log_entries: Iterable[str] = (),
305    log_axioms_file: Path | None = None,
306    log_succ_rewrites: bool = True,
307    log_fail_rewrites: bool = False,
308    start_server: bool = True,
309    fallback_on: Iterable[FallbackReason] | None = None,
310    interim_simplification: int | None = None,
311    simplify_each: int | None = None,
312    no_post_exec_simplify: bool = False,
313) -> Iterator[CTermSymbolic]:
314    if start_server:
315        # Old way of handling KoreServer, to be removed
316        with kore_server(
317            definition_dir=definition_dir,
318            llvm_definition_dir=llvm_definition_dir,
319            module_name=definition.main_module_name,
320            port=port,
321            command=kore_rpc_command,
322            bug_report=bug_report,
323            smt_timeout=smt_timeout,
324            smt_retry_limit=smt_retry_limit,
325            smt_tactic=smt_tactic,
326            haskell_log_format=haskell_log_format,
327            haskell_log_entries=haskell_log_entries,
328            log_axioms_file=log_axioms_file,
329            fallback_on=fallback_on,
330            interim_simplification=interim_simplification,
331            simplify_each=simplify_each,
332            no_post_exec_simplify=no_post_exec_simplify,
333        ) as server:
334            with KoreClient('localhost', server.port, bug_report=bug_report, bug_report_id=id) as client:
335                yield CTermSymbolic(
336                    client, definition, log_succ_rewrites=log_succ_rewrites, log_fail_rewrites=log_fail_rewrites
337                )
338    else:
339        if port is None:
340            raise ValueError('Missing port with start_server=False')
341        with KoreClient('localhost', port, bug_report=bug_report, bug_report_id=id) as client:
342            yield CTermSymbolic(
343                client, definition, log_succ_rewrites=log_succ_rewrites, log_fail_rewrites=log_fail_rewrites
344            )
