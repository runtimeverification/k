56def _gen_modules(
57    library_name: str,
58    genmodel: dict[str, tuple[Callable[[], Module], list[str]]],
59) -> dict[Path, Module]:
60    def gen_module(generator: Callable[[], Module], imports: Iterable[str]) -> Module:
61        imports = tuple(f'{library_name}.{importt}' for importt in imports)
62        module = generator()
63        module = Module(imports=imports, commands=module.commands)
64        return module
65
66    return {
67        Path(library_name) / f'{name}.lean': gen_module(generator, imports)
68        for name, (generator, imports) in genmodel.items()
69    }

72def _write_modules(output_dir: Path, modules: dict[Path, Module]) -> None:
73    for path, module in modules.items():
74        (output_dir / path).write_text(str(module))
