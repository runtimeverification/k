 12def indent(text: str, n: int) -> str:
 13    indent = n * ' '
 14    res = []
 15    for line in text.splitlines():
 16        res.append(f'{indent}{line}' if line else '')
 17    return '\n'.join(res)

114
115    def __init__(
116        self,
117        ident: str | DeclId,
118        signature: Signature | None = None,
119        ctors: Iterable[Ctor] | None = None,
120        deriving: Iterable[str] | None = None,
121        modifiers: Modifiers | None = None,
122    ):
123        ident = DeclId(ident) if isinstance(ident, str) else ident
124        ctors = tuple(ctors) if ctors is not None else ()
125        deriving = tuple(deriving) if deriving is not None else ()
126        object.__setattr__(self, 'ident', ident)

128        object.__setattr__(self, 'ctors', ctors)

135        where = ' where' if self.ctors else ''
136        deriving = ', '.join(self.deriving)
137
138        lines = []
139        lines.append(f'{modifiers}inductive {self.ident}{signature}{where}')
140        for ctor in self.ctors:

142        if deriving:
143            lines.append(f'  deriving {deriving}')
144        return '\n'.join(lines)

149class Instance(Declaration):
150    signature: Signature
151    val: DeclVal
152    attr_kind: AttrKind | None
153    priority: int | None
154    ident: DeclId | None
155    modifiers: Modifiers | None
156
157    def __init__(
158        self,
159        signature: Signature,
160        val: DeclVal,
161        attr_kind: AttrKind | None = None,
162        priority: int | None = None,
163        ident: str | DeclId | None = None,
164        modifiers: Modifiers | None = None,
165    ):
166        if priority and priority < 0:
167            raise ValueError('Priority must be non-negative')
168        if not signature.ty:
169            # TODO refine type to avoid this check
170            raise ValueError('Missing type from signature')
171        ident = DeclId(ident) if isinstance(ident, str) else ident
172        object.__setattr__(self, 'signature', signature)
173        object.__setattr__(self, 'val', val)
174        object.__setattr__(self, 'attr_kind', attr_kind)
175        object.__setattr__(self, 'priority', priority)
176        object.__setattr__(self, 'ident', ident)
177        object.__setattr__(self, 'modifiers', modifiers)
178
179    def __str__(self) -> str:
180        modifiers = f'{self.modifiers} ' if self.modifiers else ''
181        attr_kind = f'{self.attr_kind.value} ' if self.attr_kind else ''
182        priority = f' (priority := {self.priority})' if self.priority is not None else ''
183        ident = f' {self.ident}' if self.ident else ''
184        signature = f' {self.signature}' if self.signature else ''
185
186        decl = f'{modifiers}{attr_kind}instance{priority}{ident}{signature}'
187
188        match self.val:
189            case SimpleVal():
190                return f'{decl} := {self.val}'
191            case StructVal(fields):
192                lines = []
193                lines.append(f'{decl} where')
194                lines.extend(indent(str(field), 2) for field in fields)
195                return '\n'.join(lines)
196            case _:
197                raise AssertionError()

207    def __init__(
208        self,
209        ident: str | DeclId,
210        val: DeclVal,
211        signature: Signature | None = None,
212        deriving: Iterable[str] | None = None,
213        modifiers: Modifiers | None = None,
214    ):
215        ident = DeclId(ident) if isinstance(ident, str) else ident
216        deriving = tuple(deriving) if deriving is not None else ()

351class Structure(Declaration):
352    ident: DeclId
353    signature: Signature | None
354    extends: tuple[Term, ...]
355    ctor: StructCtor | None
356    deriving: tuple[str, ...]
357    modifiers: Modifiers | None
358
359    def __init__(
360        self,
361        ident: str | DeclId,
362        signature: Signature | None = None,
363        extends: Iterable[Term] | None = None,
364        ctor: StructCtor | None = None,
365        deriving: Iterable[str] | None = None,
366        modifiers: Modifiers | None = None,
367    ):
368        ident = DeclId(ident) if isinstance(ident, str) else ident
369        extends = tuple(extends) if extends is not None else ()
370        deriving = tuple(deriving) if deriving is not None else ()
371        object.__setattr__(self, 'ident', ident)
372        object.__setattr__(self, 'signature', signature)
373        object.__setattr__(self, 'extends', extends)
374        object.__setattr__(self, 'ctor', ctor)
375        object.__setattr__(self, 'deriving', deriving)
376        object.__setattr__(self, 'modifiers', modifiers)
377
378    def __str__(self) -> str:
379        lines = []
380
381        modifiers = f'{self.modifiers} ' if self.modifiers else ''
382        binders = (
383            ' '.join(str(binder) for binder in self.signature.binders)
384            if self.signature and self.signature.binders
385            else ''
386        )
387        binders = f' {binders}' if binders else ''
388        extends = ', '.join(str(extend) for extend in self.extends)
389        extends = f' extends {extends}' if extends else ''
390        ty = f' : {self.signature.ty}' if self.signature and self.signature.ty else ''
391        where = ' where' if self.ctor else ''
392        lines.append(f'{modifiers}structure {self.ident}{binders}{extends}{ty}{where}')
393
394        if self.ctor:
395            lines.extend(f'  {line}' for line in str(self.ctor).splitlines())
396
397        if self.deriving:
398            deriving = ', '.join(self.deriving)
399            lines.append(f'  deriving {deriving}')
400
401        return '\n'.join(lines)

560class Modifiers:
561    attrs: tuple[Attr, ...]
562    visibility: Visibility | None
563    noncomputable: bool
564    unsafe: bool
565    totality: Totality | None
566
567    def __init__(
568        self,
569        *,
570        attrs: Iterable[str | Attr] | None = None,
571        visibility: str | Visibility | None = None,
572        noncomputable: bool | None = None,
573        unsafe: bool | None = None,
574        totality: str | Totality | None = None,
575    ):
576        attrs = tuple(Attr(attr) if isinstance(attr, str) else attr for attr in attrs) if attrs is not None else ()
577        visibility = Visibility(visibility) if isinstance(visibility, str) else visibility
578        noncomputable = bool(noncomputable)
579        unsafe = bool(unsafe)
580        totality = Totality(totality) if isinstance(totality, str) else totality
581        object.__setattr__(self, 'attrs', attrs)
582        object.__setattr__(self, 'visibility', visibility)
583        object.__setattr__(self, 'noncomputable', noncomputable)
584        object.__setattr__(self, 'unsafe', unsafe)
585        object.__setattr__(self, 'totality', totality)
586
587    def __str__(self) -> str:
588        chunks = []
589        if self.attrs:
590            attrs = ', '.join(str(attr) for attr in self.attrs)
591            chunks.append(f'@[{attrs}]')
592        if self.visibility:
593            chunks.append(self.visibility.value)
594        if self.noncomputable:
595            chunks.append('noncomputable')
596        if self.unsafe:
597            chunks.append('unsafe')
598        if self.totality:
599            chunks.append(self.totality.value)
600        return ' '.join(chunks)
