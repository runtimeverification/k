124    @staticmethod
125    def from_axiom(axiom: Axiom) -> Rule:
126        if isinstance(axiom.pattern, Rewrites):
127            return RewriteRule.from_axiom(axiom)
128
129        if 'simplification' not in axiom.attrs_by_key:
130            return FunctionRule.from_axiom(axiom)
131
132        match axiom.pattern:
133            case Implies(right=Equals(left=App())):
134                return AppRule.from_axiom(axiom)
135            case Implies(right=Equals(left=Ceil())):
136                return CeilRule.from_axiom(axiom)
137            case Implies(right=Equals(left=Equals())):
138                return EqualsRule.from_axiom(axiom)
139            case _:
140                raise ValueError(f'Cannot parse simplification rule: {axiom.text}')

143    def is_rule(axiom: Axiom) -> bool:
144        if axiom == _INJ_AXIOM:

147        if axiom == _APPEND_DOTK_AXIOM:

149

152

154            return False

156        return True

249    def to_axiom(self) -> Axiom:
250        R = SortVar('R')  # noqa N806
251
252        def arg_list(rest: Pattern, arg_pair: tuple[EVar, Pattern]) -> Pattern:
253            var, arg = arg_pair
254            return And(R, (In(var.sort, R, var, arg), rest))
255
256        vars = tuple(EVar(f'X{i}', sort) for i, sort in enumerate(self.arg_sorts))
257
258        # \and{R}(\in{S1, R}(X1 : S1, Arg1), \and{R}(\in{S2, R}(X2 : S2, Arg2), \top{R}())) etc.
259        _args = reduce(
260            arg_list,
261            reversed(tuple(zip(vars, self.lhs.args, strict=True))),
262            cast('Pattern', Top(R)),
263        )
264
265        _req = _to_ml_pred(self.req, R)
266        req = And(R, (_req, _args))
267        if self.anti_left:
268            req = And(R, (Not(R, self.anti_left), req))
269
270        app = self.lhs.let(args=vars)
271        ens = _to_ml_pred(self.ens, self.sort)
272
273        return Axiom(
274            (R,),
275            Implies(
276                R,
277                req,
278                Equals(self.sort, R, app, And(self.sort, (self.rhs, ens))),
279            ),
280        )

320    @staticmethod
321    def _extract_args(pattern: Pattern) -> tuple[tuple[Sort, ...], tuple[Pattern, ...]]:
322        match pattern:
323            case Top():
324                return (), ()
325            case And(ops=(In(left=EVar(sort=sort), right=arg), rest)):
326                sorts, args = FunctionRule._extract_args(rest)
327                return (sort,) + sorts, (arg,) + args
328            case _:
329                raise ValueError(f'Cannot extract argument list from pattern: {pattern.text}')

336    def to_axiom(self) -> Axiom:
337        R = SortVar('R')  # noqa N806
338
339        vars = (R, self.sort) if isinstance(self.sort, SortVar) else (R,)
340        req = _to_ml_pred(self.req, R)
341        ens = _to_ml_pred(self.ens, self.sort)
342
343        return Axiom(
344            vars,
345            Implies(
346                R,
347                req,
348                Equals(self.sort, R, self.lhs, And(self.sort, (self.rhs, ens))),
349            ),
350            attrs=(
351                App(
352                    'simplification',
353                    args=() if self.priority == 50 else (String(str(self.priority)),),
354                ),
355            ),
356        )

358    @staticmethod
359    def _extract(axiom: Axiom, lhs_type: type[P]) -> tuple[P, Pattern, Pattern | None, Pattern | None, Sort]:
360        match axiom.pattern:
361            case Implies(left=_req, right=Equals(op_sort=sort, left=lhs, right=_rhs)):
362                req = _extract_condition(_req)
363                rhs, ens = _extract_rhs(_rhs)
364                if not isinstance(lhs, lhs_type):
365                    raise ValueError(f'Invalid LHS type from simplification axiom: {axiom.text}')
366                return lhs, rhs, req, ens, sort
367            case _:
368                raise ValueError(f'Cannot extract simplification rule from axiom: {axiom.text}')

461def _extract_rhs(pattern: Pattern) -> tuple[Pattern, Pattern | None]:
462    match pattern:
463        case And(ops=(rhs, _ens)):
464            return rhs, _extract_condition(_ens)
465        case _:
466            raise ValueError(f'Cannot extract RHS from pattern: {pattern.text}')

469def _extract_condition(pattern: Pattern) -> Pattern | None:
470    match pattern:
471        case Top():
472            return None
473        case Equals(left=cond, right=DV(SortApp('SortBool'), String('true'))):
474            return cond
475        case _:
476            raise ValueError(f'Cannot extract condition from pattern: {pattern.text}')

488def _extract_label(axiom: Axiom) -> str | None:
489    attrs = axiom.attrs_by_key

499def _extract_priority(axiom: Axiom) -> int:
500    attrs = axiom.attrs_by_key
501    match attrs.get('priority'):
502        case App(args=(String(p),)):
503            assert 'owise' not in attrs
504            return int(p)
505        case None:
506            return 200 if 'owise' in attrs else 50
507        case _:
508            raise ValueError(f'Cannot extract priority from axiom: {axiom.text}')

511def _extract_simpl_priority(axiom: Axiom) -> int:
512    attrs = axiom.attrs_by_key
513    match attrs['simplification']:
514        case App(args=() | (String(''),)):
515            return 50
516        case App(args=(String(p),)):
517            return int(p)
518        case _:
519            raise ValueError(f'Cannot extract simplification priority from axiom: {axiom.text}')

522def _to_ml_pred(pattern: Pattern | None, sort: Sort) -> Pattern:
523    if pattern is None:
524        return Top(sort)
525
526    return Equals(BOOL, sort, pattern, TRUE)
