360    @property
361    def step_event(self) -> LLVMStepEvent:
362        """Returns the LLVMStepEvent associated with the argument if any."""
363        if isinstance(self._argument.step_event, llvm_rule_event):
364            return LLVMRuleEvent(self._argument.step_event)
365        elif isinstance(self._argument.step_event, llvm_side_condition_event):
366            return LLVMSideConditionEventEnter(self._argument.step_event)
367        elif isinstance(self._argument.step_event, llvm_side_condition_end_event):
368            return LLVMSideConditionEventExit(self._argument.step_event)
369        elif isinstance(self._argument.step_event, llvm_function_event):
370            return LLVMFunctionEvent(self._argument.step_event)
371        elif isinstance(self._argument.step_event, llvm_function_exit_event):
372            return LLVMFunctionExitEvent(self._argument.step_event)
373        elif isinstance(self._argument.step_event, llvm_hook_event):
374            return LLVMHookEvent(self._argument.step_event)
375        elif isinstance(self._argument.step_event, llvm_pattern_matching_failure_event):
376            return LLVMPatternMatchingFailureEvent(self._argument.step_event)
377        else:
378            raise AssertionError()

571    def __iter__(self) -> Iterator[LLVMEventAnnotated]:
572        """Yield LLVMEventAnnotated options.
573
574        This method is an iterator that yields LLVMEventAnnotated options.
575        It iterates over the events in the trace and returns the next event as an LLVMEventAnnotated object.
576
577        Yields:
578            LLVMEventAnnotated: The next LLVMEventAnnotated option.
579        """
580        while True:
581            next_event = self._rewrite_trace_iterator.get_next_event()
582            if next_event is None:
583                return
584            else:
585                yield LLVMEventAnnotated(next_event)

587    def __next__(self) -> LLVMEventAnnotated:
588        """Yield the next LLVMEventAnnotated object from the iterator.
589
590        Returns:
591            LLVMEventAnnotated: The next LLVMEventAnnotated object.
592
593        Raises:
594            StopIteration: If there are no more events in the iterator.
595        """
596        next_event = self._rewrite_trace_iterator.get_next_event()
597        if next_event is not None:
