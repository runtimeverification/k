 42    def is_explorable(self, node_id: NodeIdLike) -> bool:
 43        return (
 44            self.kcfg.is_leaf(node_id)
 45            and not self.is_terminal(node_id)
 46            and not self.kcfg.is_stuck(node_id)
 47            and not self.kcfg.is_vacuous(node_id)
 48        )
