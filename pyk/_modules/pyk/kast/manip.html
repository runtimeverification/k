 44def is_term_like(kast: KInner) -> bool:
 45    non_term_found = False
 46
 47    def _is_term_like(_kast: KInner) -> None:
 48        nonlocal non_term_found
 49        match _kast:
 50            case KVariable(name, _):
 51                if name.startswith('@'):
 52                    non_term_found = True
 53            case KApply(KLabel(name, _), _):
 54                if name in {
 55                    '#Equals',
 56                    '#And',
 57                    '#Or',
 58                    '#Top',
 59                    '#Bottom',
 60                    '#Implies',
 61                    '#Not',
 62                    '#Ceil',
 63                    '#Forall',
 64                    '#Exists',
 65                }:
 66                    non_term_found = True
 67
 68    collect(_is_term_like, kast)
 69    return not non_term_found

 72def sort_assoc_label(label: str, kast: KInner) -> KInner:
 73    res: KInner | None = None
 74    if type(kast) is KApply and kast.label.name == label:
 75        terms = sorted(flatten_label(label, kast))
 76        for term in reversed(terms):
 77            if not res:
 78                res = term
 79            else:
 80                res = kast.label(term, res)
 81        assert res is not None
 82        return res
 83    return kast

117def ml_pred_to_bool(kast: KInner, unsafe: bool = False) -> KInner:
118    def _ml_constraint_to_bool(_kast: KInner) -> KInner:
119        if type(_kast) is KApply:
120            if _kast.label.name == '#Top':
121                return TRUE
122            if _kast.label.name == '#Bottom':
123                return FALSE
124            if _kast.label.name == '#Not' and len(_kast.args) == 1:
125                return notBool(_ml_constraint_to_bool(_kast.args[0]))
126            if _kast.label.name == '#And':
127                return andBool(map(_ml_constraint_to_bool, _kast.args))
128            if _kast.label.name == '#Or':
129                return orBool(map(_ml_constraint_to_bool, _kast.args))
130            if _kast.label.name == '#Implies' and len(_kast.args) == 2:
131                return impliesBool(_ml_constraint_to_bool(_kast.args[0]), _ml_constraint_to_bool(_kast.args[1]))
132            if _kast.label.name == '#Equals':
133                first, second = _kast.args
134                if first == TRUE:
135                    return second
136                if first == FALSE:
137                    return notBool(second)
138                if second == TRUE:
139                    return first
140                if second == FALSE:
141                    return notBool(first)
142                if isinstance(first, (KVariable, KToken)):
143                    if first.sort == KSort('Int'):
144                        return KApply('_==Int_', _kast.args)
145                    else:
146                        return KApply('_==K_', _kast.args)
147                if isinstance(second, (KVariable, KToken)):
148                    if second.sort == KSort('Int'):
149                        return KApply('_==Int_', _kast.args)
150                    else:
151                        return KApply('_==K_', _kast.args)
152                if type(first) is KSequence and type(second) is KSequence:
153                    if first.arity == 1 and second.arity == 1:
154                        return KApply('_==K_', (first.items[0], second.items[0]))
155                if is_term_like(first) and is_term_like(second):
156                    return KApply('_==K_', first, second)
157            if unsafe:
158                if _kast.label.name == '#Equals':
159                    return KApply('_==K_', _kast.args)
160                if _kast.label.name == '#Ceil':
161                    ceil_var = abstract_term_safely(_kast, base_name='Ceil')
162                    _LOGGER.warning(f'Converting #Ceil condition to variable {ceil_var.name}: {_kast}')
163                    return ceil_var
164                if _kast.label.name == '#Exists':
165                    exists_var = abstract_term_safely(_kast, base_name='Exists')
166                    _LOGGER.warning(f'Converting #Exists condition to variable {exists_var.name}: {_kast}')
167                    return exists_var
168        raise ValueError(f'Could not convert ML predicate to sort Bool: {_kast}')
169
170    return _ml_constraint_to_bool(kast)

173def simplify_bool(k: KInner) -> KInner:
174    if k is None:
175        return None
176
177    # fmt: off
178    simplify_rules = [ (KApply('_==K_', [KVariable('#LHS'), TRUE]), KVariable('#LHS'))
179                     , (KApply('_==K_', [TRUE, KVariable('#RHS')]), KVariable('#RHS'))
180                     , (KApply('_==K_', [KVariable('#LHS'), FALSE]), notBool(KVariable('#LHS')))
181                     , (KApply('_==K_', [FALSE, KVariable('#RHS')]), notBool(KVariable('#RHS')))
182                     , (notBool(FALSE), TRUE)
183                     , (notBool(TRUE), FALSE)
184                     , (notBool(KApply('_==K_'    , [KVariable('#V1'), KVariable('#V2')])), KApply('_=/=K_'   , [KVariable('#V1'), KVariable('#V2')]))
185                     , (notBool(KApply('_=/=K_'   , [KVariable('#V1'), KVariable('#V2')])), KApply('_==K_'    , [KVariable('#V1'), KVariable('#V2')]))
186                     , (notBool(KApply('_==Int_'  , [KVariable('#V1'), KVariable('#V2')])), KApply('_=/=Int_' , [KVariable('#V1'), KVariable('#V2')]))
187                     , (notBool(KApply('_=/=Int_' , [KVariable('#V1'), KVariable('#V2')])), KApply('_==Int_'  , [KVariable('#V1'), KVariable('#V2')]))
188                     , (andBool([TRUE, KVariable('#REST')]), KVariable('#REST'))
189                     , (andBool([KVariable('#REST'), TRUE]), KVariable('#REST'))
190                     , (andBool([FALSE, KVariable('#REST')]), FALSE)
191                     , (andBool([KVariable('#REST'), FALSE]), FALSE)
192                     , (orBool([FALSE, KVariable('#REST')]), KVariable('#REST'))
193                     , (orBool([KVariable('#REST'), FALSE]), KVariable('#REST'))
194                     , (orBool([TRUE, KVariable('#REST')]), TRUE)
195                     , (orBool([KVariable('#REST'), TRUE]), TRUE)
196                     ]
197    # fmt: on
198
199    new_k = k
200    for rule in simplify_rules:
201        rewrite = KRewrite(*rule)
202        new_k = rewrite(new_k)
203    return new_k

218def extract_subst(term: KInner) -> tuple[Subst, KInner]:
219    subst = {}
220    rem_conjuncts: list[KInner] = []
221
222    def _extract_subst(term: KInner, term2: KInner) -> tuple[str, KInner] | None:
223        if (
224            (isinstance(term, KVariable) and term.name not in subst)
225            and not (isinstance(term2, KVariable) and term2.name in subst)
226            and term.name not in free_vars(term2)
227        ):
228            return (term.name, term2)
229        if (
230            (isinstance(term2, KVariable) and term2.name not in subst)
231            and not (isinstance(term, KVariable) and term.name in subst)
232            and term2.name not in free_vars(term)
233        ):
234            return (term2.name, term)
235        if term == TRUE and isinstance(term2, KApply) and term2.label.name in {'_==K_', '_==Int_'}:
236            return _extract_subst(term2.args[0], term2.args[1])
237        if term2 == TRUE and isinstance(term, KApply) and term.label.name in {'_==K_', '_==Int_'}:
238            return _extract_subst(term.args[0], term.args[1])
239        return None
240
241    for conjunct in flatten_label('#And', term):
242        if isinstance(conjunct, KApply) and conjunct.label.name == '#Equals':
243            if _conjunct_subst := _extract_subst(conjunct.args[0], conjunct.args[1]):
244                name, value = _conjunct_subst
245                subst[name] = value
246            else:
247                rem_conjuncts.append(conjunct)
248        else:
249            rem_conjuncts.append(conjunct)
250
251    return Subst(subst), mlAnd(rem_conjuncts)

266def propagate_up_constraints(k: KInner) -> KInner:
267    def _propagate_up_constraints(_k: KInner) -> KInner:
268        if not (type(_k) is KApply and _k.label.name == '#Or'):
269            return _k
270        top_sort = _k.label.params[0]
271        conjuncts1 = flatten_label('#And', _k.args[0])
272        conjuncts2 = flatten_label('#And', _k.args[1])
273        (common1, l1, r1) = find_common_items(conjuncts1, conjuncts2)
274        (common2, r2, l2) = find_common_items(r1, l1)
275        common = common1 + common2
276        if len(common) == 0:
277            return _k
278        conjunct1 = mlAnd(l2, sort=top_sort)
279        conjunct2 = mlAnd(r2, sort=top_sort)
280        disjunct = mlOr([conjunct1, conjunct2], sort=top_sort)
281        return mlAnd([disjunct] + common, sort=top_sort)
282
283    return bottom_up(_propagate_up_constraints, k)

286def split_config_and_constraints(kast: KInner) -> tuple[KInner, KInner]:
287    conjuncts = flatten_label('#And', kast)
288    term = None
289    constraints = []
290    for c in conjuncts:
291        if type(c) is KApply and c.is_cell:
292            if term:
293                raise ValueError(f'Found two configurations in pattern:\n\n{term}\n\nand\n\n{c}')
294            term = c
295        else:
296            constraints.append(c)
297    if not term:
298        raise ValueError(f'Could not find configuration for: {kast}')
299    return (term, mlAnd(constraints, GENERATED_TOP_CELL))

307def split_config_from(configuration: KInner) -> tuple[KInner, dict[str, KInner]]:
308    """Split the substitution from a given configuration.
309
310    Given an input configuration `config`, will return a tuple `(symbolic_config, subst)`, where:
311
312        1.  `config == substitute(symbolic_config, subst)`
313        2.  `symbolic_config` is the same configuration structure, but where the contents of leaf cells is replaced with a fresh KVariable.
314        3.  `subst` is the substitution for the generated KVariables back to the original configuration contents.
315    """
316    initial_substitution = {}
317
318    def _replace_with_var(k: KInner) -> KInner:
319        if type(k) is KApply and k.is_cell:
320            if k.arity == 1 and not (type(k.args[0]) is KApply and k.args[0].is_cell):
321                config_var = cell_label_to_var_name(k.label.name)
322                initial_substitution[config_var] = k.args[0]
323                return KApply(k.label, [KVariable(config_var)])
324        return k
325
326    symbolic_config = top_down(_replace_with_var, configuration)
327    return (symbolic_config, initial_substitution)

330def collapse_dots(kast: KInner) -> KInner:
331    """Given a configuration with structural frames `...`, minimize the structural frames needed.
332
333    Args:
334        kast: A configuration, potentially with structural frames.
335
336    Returns:
337        The same configuration, with the amount of structural framing minimized.
338    """
339
340    def _collapse_dots(_kast: KInner) -> KInner:
341        if type(_kast) is KApply:
342            if _kast.is_cell and _kast.arity == 1 and _kast.args[0] == DOTS:
343                return DOTS
344            new_args = [arg for arg in _kast.args if arg != DOTS]
345            if _kast.is_cell and len(new_args) == 0:
346                return DOTS
347            if len(new_args) < len(_kast.args):
348                new_args.append(DOTS)
349            return _kast.let(args=new_args)
350        elif type(_kast) is KRewrite:
351            if _kast.lhs == DOTS:
352                return DOTS
353        return _kast
354
355    return bottom_up(_collapse_dots, kast)

358def push_down_rewrites(kast: KInner) -> KInner:
359    def _push_down_rewrites(_kast: KInner) -> KInner:
360        if type(_kast) is KRewrite:
361            lhs = _kast.lhs
362            rhs = _kast.rhs
363            if lhs == rhs:
364                return lhs
365            if type(lhs) is KVariable and type(rhs) is KVariable and lhs.name == rhs.name:
366                return lhs
367            if type(lhs) is KApply and type(rhs) is KApply and lhs.label == rhs.label and lhs.arity == rhs.arity:
368                new_args = [
369                    KRewrite(left_arg, right_arg) for left_arg, right_arg in zip(lhs.args, rhs.args, strict=True)
370                ]
371                return lhs.let(args=new_args)
372            if type(lhs) is KSequence and type(rhs) is KSequence and lhs.arity > 0 and rhs.arity > 0:
373                if lhs.arity == 1 and rhs.arity == 1:
374                    return KRewrite(lhs.items[0], rhs.items[0])
375                if lhs.items[0] == rhs.items[0]:
376                    lower_rewrite = _push_down_rewrites(KRewrite(KSequence(lhs.items[1:]), KSequence(rhs.items[1:])))
377                    return KSequence([lhs.items[0], lower_rewrite])
378                if lhs.items[-1] == rhs.items[-1]:
379                    lower_rewrite = _push_down_rewrites(
380                        KRewrite(KSequence(lhs.items[0:-1]), KSequence(rhs.items[0:-1]))
381                    )
382                    return KSequence([lower_rewrite, lhs.items[-1]])
383            if (
384                type(lhs) is KSequence
385                and lhs.arity > 0
386                and type(lhs.items[-1]) is KVariable
387                and type(rhs) is KVariable
388                and lhs.items[-1] == rhs
389            ):
390                return KSequence([KRewrite(KSequence(lhs.items[0:-1]), KSequence([])), rhs])
391        return _kast
392
393    return top_down(_push_down_rewrites, kast)

434def useless_vars_to_dots(kast: KInner, keep_vars: Iterable[str] = ()) -> KInner:
435    """Structurally abstract away useless variables.
436
437    Args:
438        kast: A term.
439        keep_vars: Iterable of variables to keep.
440
441    Returns:
442        The term with the useless varables structurally abstracted.
443    """
444    num_occs = count_vars(kast) + Counter(keep_vars)
445
446    def _collapse_useless_vars(_kast: KInner) -> KInner:
447        if type(_kast) is KApply and _kast.is_cell:
448            new_args: list[KInner] = []
449            for arg in _kast.args:
450                if type(arg) is KVariable and num_occs[arg.name] == 1:
451                    new_args.append(DOTS)
452                else:
453                    new_args.append(arg)
454            return _kast.let(args=new_args)
455        return _kast
456
457    return bottom_up(_collapse_useless_vars, kast)

509def minimize_term(term: KInner, keep_vars: Iterable[str] = ()) -> KInner:
510    """Minimize a K term for pretty-printing.
511
512    - Variables only used once will be removed.
513    - Unused cells will be abstracted.
514    - Useless conditions will be attempted to be removed.
515
516    Args:
517        kast: A term.
518
519    Returns:
520        The term, minimized.
521    """
522    term = inline_cell_maps(term)
523    term = remove_semantic_casts(term)
524    term = useless_vars_to_dots(term, keep_vars=keep_vars)
525    term = collapse_dots(term)
526    return term

529def minimize_rule_like(rule: RL, keep_vars: Iterable[str] = ()) -> RL:
530    """Minimize a K rule or claim for pretty-printing.
531
532    - Variables only used once will be removed.
533    - Unused cells will be abstracted.
534    - Useless side-conditions will be attempted to be removed.
535
536    Args:
537        rule: A K rule or claim.
538
539    Returns:
540        The rule or claim, minimized.
541    """
542    body = rule.body
543    requires = rule.requires
544    ensures = rule.ensures
545
546    requires = andBool(flatten_label('_andBool_', requires))
547    requires = simplify_bool(requires)
548
549    ensures = andBool(flatten_label('_andBool_', ensures))
550    ensures = simplify_bool(ensures)
551
552    constrained_vars = set(keep_vars) | free_vars(requires) | free_vars(ensures)
553    body = minimize_term(body, keep_vars=constrained_vars)
554
555    return rule.let(body=body, requires=requires, ensures=ensures)

595def abstract_term_safely(
596    kast: KInner, base_name: str = 'V', sort: KSort | None = None, existing_var_names: set[str] | None = None
597) -> KVariable:
598    def _abstract(k: KInner) -> KVariable:
599        vname = hash_str(k)[0:8]
600        return KVariable(base_name + '_' + vname, sort=sort)
601
602    new_var = _abstract(kast)
603    if existing_var_names is not None:
604        while new_var.name in existing_var_names:
605            new_var = _abstract(new_var)
606    return new_var

609def apply_existential_substitutions(state: KInner, constraints: Iterable[KInner]) -> tuple[KInner, Iterable[KInner]]:
610    pattern = mlEqualsTrue(KApply('_==K_', [KVariable('#VAR'), KVariable('#VAL')]))
611    subst = {}
612    new_constraints = []
613    for c in constraints:
614        match = pattern.match(c)
615        if match is not None and type(match['#VAR']) is KVariable and match['#VAR'].name.startswith('?'):
616            subst[match['#VAR'].name] = match['#VAL']
617        else:
618            new_constraints.append(c)
619    return (Subst(subst)(state), [Subst(subst)(c) for c in new_constraints])

622def undo_aliases(definition: KDefinition, kast: KInner) -> KInner:
623    aliases = []
624    for rule in definition.alias_rules:
625        rewrite = rule.body
626        if type(rewrite) is not KRewrite:
627            raise ValueError(f'Expected KRewrite as alias body, found: {rewrite}')
628        if rule.requires is not None and rule.requires != TRUE:
629            raise ValueError(f'Expended empty requires clause on alias, found: {rule.requires}')
630        if rule.ensures is not None and rule.ensures != TRUE:
631            raise ValueError(f'Expended empty ensures clause on alias, found: {rule.ensures}')
632        aliases.append(KRewrite(rewrite.rhs, rewrite.lhs))
633    return indexed_rewrite(kast, aliases)

636def rename_generated_vars(term: KInner) -> KInner:
637    vars: set[str] = set(free_vars(term))
638    cell_stack: list[str] = []
639
640    def _rename_vars(k: KInner) -> KInner:
641        if type(k) is KApply and k.is_cell:
642            cell_stack.append(cell_label_to_var_name(k.label.name))
643            res = k.map_inner(_rename_vars)
644            cell_stack.pop()
645            return res
646
647        if type(k) is KVariable and k.name.startswith(('_Gen', '?_Gen', '_DotVar', '?_DotVar')):
648            if not cell_stack:
649                return k
650            cell_name = cell_stack[-1]
651            new_var = abstract_term_safely(k, base_name=cell_name, sort=k.sort, existing_var_names=vars)
652            vars.add(new_var.name)
653            return new_var
654
655        return k.map_inner(_rename_vars)
656
657    return _rename_vars(term)

675def remove_useless_constraints(constraints: Iterable[KInner], initial_vars: Iterable[str]) -> list[KInner]:
676    """Remove constraints that do not depend on a given iterable of variables (directly or indirectly).
677
678    Args:
679        constraints: Iterable of constraints to filter.
680        initial_vars: Initial iterable of variables to keep constraints for.
681
682    Returns:
683        A list of constraints with only those constraints that contain the initial variables,
684        or variables that depend on those through other constraints in the list.
685    """
686    used_vars = list(initial_vars)
687    prev_len_used_vars = 0
688    new_constraints = []
689    while len(used_vars) > prev_len_used_vars:
690        prev_len_used_vars = len(used_vars)
691        for c in constraints:
692            if c not in new_constraints:
693                new_vars = free_vars(c)
694                if any(v in used_vars for v in new_vars):
695                    new_constraints.append(c)
696                    used_vars.extend(new_vars)
697        used_vars = list(set(used_vars))
698    return new_constraints

734def build_rule(
735    rule_id: str,
736    init_config: KInner,
737    final_config: KInner,
738    init_constraints: Iterable[KInner] = (),
739    final_constraints: Iterable[KInner] = (),
740    priority: int | None = None,
741    keep_vars: Iterable[str] = (),
742    defunc_with: KDefinition | None = None,
743) -> tuple[KRule, Subst]:
744    """Return a `KRule` between the supplied initial and final states.
745
746    Args:
747        rule_id: Label to give the rule.
748        init_config: State to put on LHS of the rule.
749        final_config: State to put on RHS of the rule.
750        init_constraints: Constraints to use as `requires` clause.
751        final_constraints: Constraints to use as `ensures` clause.
752        priority: Priority index to assign to generated rules.
753        keep_vars: Variables to leave in the side-conditions even if not bound in the configuration.
754        defunc_with: KDefinition for filtering out function symbols on LHS of rules.
755
756    Returns:
757        A tuple ``(rule, var_map)`` where
758
759        - ``rule``: A `KRule` with variable naming conventions applied
760          so that it should be parseable by the K Frontend.
761        - ``var_map``: The variable renamings applied to make the rule parseable by the K Frontend
762          (which can be undone to recover the original variables).
763    """
764    init_constraints = [normalize_ml_pred(c) for c in init_constraints]
765    final_constraints = [normalize_ml_pred(c) for c in final_constraints]
766    final_constraints = [c for c in final_constraints if c not in init_constraints]
767    if defunc_with is not None:
768        init_config, new_constraints = defunctionalize(defunc_with, init_config)
769        init_constraints = init_constraints + new_constraints
770    init_term = mlAnd([init_config] + init_constraints)
771    final_term = mlAnd([final_config] + final_constraints)
772
773    lhs_vars = free_vars(init_term)
774    rhs_vars = free_vars(final_term)
775    occurrences = var_occurrences(
776        mlAnd(
777            [push_down_rewrites(KRewrite(init_config, final_config))] + init_constraints + final_constraints,
778            GENERATED_TOP_CELL,
779        )
780    )
781    sorted_vars = keep_vars_sorted(occurrences)
782    v_subst: dict[str, KVariable] = {}
783    vremap_subst: dict[str, KVariable] = {}
784    for v in occurrences:
785        new_v = v
786        if len(occurrences[v]) == 1:
787            new_v = '_' + new_v
788        if v in rhs_vars and v not in lhs_vars:
789            new_v = '?' + new_v
790        if new_v != v:
791            v_subst[v] = KVariable(new_v, sorted_vars[v].sort)
792            vremap_subst[new_v] = sorted_vars[v]
793
794    new_init_config = Subst(v_subst)(init_config)
795    new_init_constraints = [Subst(v_subst)(c) for c in init_constraints]
796    new_final_config, new_final_constraints = apply_existential_substitutions(
797        Subst(v_subst)(final_config), [Subst(v_subst)(c) for c in final_constraints]
798    )
799
800    rule_body = push_down_rewrites(KRewrite(new_init_config, new_final_config))
801    rule_requires = simplify_bool(ml_pred_to_bool(mlAnd(new_init_constraints)))
802    rule_ensures = simplify_bool(ml_pred_to_bool(mlAnd(new_final_constraints)))
803    att_entries = [] if priority is None else [Atts.PRIORITY(str(priority))]
804    rule_att = KAtt(entries=att_entries)
805
806    rule = KRule(rule_body, requires=rule_requires, ensures=rule_ensures, att=rule_att)
807    rule = rule.update_atts([Atts.LABEL(rule_id)])
808
809    return (rule, Subst(vremap_subst))

821def no_cell_rewrite_to_dots(term: KInner) -> KInner:
822    """Transform a given term by replacing the contents of each cell with dots if the LHS and RHS are the same.
823
824    This function recursively traverses the cells in a term.
825    When it finds a cell whose left-hand side (LHS) is identical to its right-hand side (RHS),
826    it replaces the cell's contents with a predefined DOTS.
827
828    Args:
829        term: The term to be transformed.
830
831    Returns:
832        The transformed term, where specific cell contents have been replaced with dots.
833    """
834
835    def _no_cell_rewrite_to_dots(_term: KInner) -> KInner:
836        if type(_term) is KApply and _term.is_cell:
837            lhs = extract_lhs(_term)
838            rhs = extract_rhs(_term)
839            if lhs == rhs:
840                return KApply(_term.label, [DOTS])
841        return _term
842
843    config, _subst = split_config_from(term)
844    subst = Subst({cell_name: _no_cell_rewrite_to_dots(cell_contents) for cell_name, cell_contents in _subst.items()})
845
846    return subst(config)

849def defunctionalize(defn: KDefinition, kinner: KInner) -> tuple[KInner, list[KInner]]:
850    """Turn non-constructor arguments into side-conditions so that a term is only constructor-like.
851
852    Args:
853        defn: The definition to pull function label information from.
854        kinner: The term to defunctionalize.
855
856    Returns:
857        A tuple of the defunctionalized term and the list of constraints generated.
858    """
859    function_symbols = [prod.klabel for prod in defn.functions if prod.klabel is not None]
860    constraints: list[KInner] = []
861
862    def _defunctionalize(_kinner: KInner) -> KInner:
863        if type(_kinner) is KApply and _kinner.label in function_symbols:
864            sort = defn.sort(_kinner)
865            assert sort is not None
866            new_var = abstract_term_safely(_kinner, base_name='F', sort=sort)
867            var_constraint = mlEquals(new_var, _kinner, arg_sort=sort)
868            constraints.append(var_constraint)
869            return new_var
870        return _kinner
871
872    new_kinner = top_down(_defunctionalize, kinner)
873    return (new_kinner, list(unique(constraints)))
