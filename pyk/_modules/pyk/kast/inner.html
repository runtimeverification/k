141    @staticmethod
142    def from_dict(dct: Mapping[str, Any]) -> KInner:
143        """Deserialize a given `KInner` into a more specific type from a dictionary."""
144        stack: list = [dct, KInner._extract_dicts(dct), []]
145        while True:
146            terms = stack[-1]
147            dcts = stack[-2]
148            dct = stack[-3]
149            idx = len(terms) - len(dcts)
150            if not idx:
151                stack.pop()
152                stack.pop()
153                stack.pop()
154                cls = globals()[dct['node']]
155                term = cls._from_dict(dct, terms)
156                if not stack:
157                    return term
158                stack[-1].append(term)
159            else:
160                dct = dcts[idx]
161                stack.append(dct)
162                stack.append(KInner._extract_dicts(dct))
163                stack.append([])

165    @staticmethod
166    def _extract_dicts(dct: Mapping[str, Any]) -> list[Mapping[str, Any]]:
167        match dct['node']:
168            case 'KApply':
169                return dct['args']
170            case 'KSequence':
171                return dct['items']
172            case 'KRewrite':
173                return [dct['lhs'], dct['rhs']]
174            case 'KAs':
175                return [dct['pattern'], dct['alias']]
176            case _:
177                return []

213        def combine(subst1: Subst | None, subst2: Subst | None) -> Subst | None:

645    def match(self, term: KInner) -> Subst | None:
646        if type(term) is KSequence:
647            if term.arity == self.arity:
648                return KInner._combine_matches(
649                    item.match(term_item) for item, term_item in zip(self.items, term.items, strict=True)
650                )
651            if 0 < self.arity and self.arity < term.arity and type(self.items[-1]) is KVariable:
652                common_length = len(self.items) - 1
653                _subst: Subst | None = Subst({self.items[-1].name: KSequence(term.items[common_length:])})
654                for si, ti in zip(self.items[:common_length], term.items[:common_length], strict=True):
655                    _subst = KInner._combine_matches([_subst, si.match(ti)])
656                return _subst
657        _LOGGER.debug(f'Matching failed: ({self}.match({term}))')
658        return None

704    def compose(self, other: Subst) -> Subst:
705        """Union two substitutions together, preferring the assignments in `self` if present in both."""
706        from_other = ((k, self(v)) for k, v in other.items())

710    def union(self, other: Subst) -> Subst | None:
711        """Union two substitutions together, failing with `None` if there are conflicting assignments."""
712        subst = dict(self)
713        for v in other:
714            if v in subst and subst[v] != other[v]:
715                return None
716            subst[v] = other[v]
717        return Subst(subst)

741        subst: dict[str, KInner] = {}

768def bottom_up_with_summary(f: Callable[[KInner, list[A]], tuple[KInner, A]], kinner: KInner) -> tuple[KInner, A]:
769    """Traverse a term from the bottom moving upward, collecting information about it.
770
771    Args:
772        f: Function to apply at each AST node to transform it and collect summary.
773        kinner: Term to apply this transformation to.
774
775    Returns:
776        A tuple of the transformed term and the summarized results.
777    """
778    stack: list = [kinner, [], []]
779    while True:
780        summaries = stack[-1]
781        terms = stack[-2]
782        term = stack[-3]
783        idx = len(terms) - len(term.terms)
784        if not idx:
785            stack.pop()
786            stack.pop()
787            stack.pop()
788            term, summary = f(term.let_terms(terms), summaries)
789            if not stack:
790                return term, summary
791            stack[-1].append(summary)
792            stack[-2].append(term)
793        else:
794            stack.append(term.terms[idx])
795            stack.append([])
796            stack.append([])

878def keep_vars_sorted(occurrences: dict[str, list[KVariable]]) -> dict[str, KVariable]:
879    """Keep the sort of variables from the occurrences dictionary."""
880    occurrences_sorted: dict[str, KVariable] = {}
881    for k, vs in occurrences.items():
882        sort = None
883        for v in vs:
884            if v.sort is not None:
885                if sort is None:
886                    sort = v.sort
887                elif sort != v.sort:
888                    sort = None
889                    break
890        occurrences_sorted[k] = KVariable(k, sort)

908def build_assoc(unit: KInner, label: str | KLabel, terms: Iterable[KInner]) -> KInner:
909    """Build an associative list.
910
911    Args:
912        unit: The empty variant of the given list type.
913        label: The associative list join operator.
914        terms: List (potentially empty) of terms to join in an associative list.
915
916    Returns:
917        The list of terms joined using the supplied label, or the unit element in the case of no terms.
918    """
919    _label = label if type(label) is KLabel else KLabel(label)
920    res: KInner | None = None
921    for term in reversed(list(terms)):
922        if term == unit:
923            continue
924        if not res:
925            res = term
926        else:
927            res = _label(term, res)
928    return res or unit

950def flatten_label(label: str, kast: KInner) -> list[KInner]:
951    """Given a cons list, return a flat Python list of the elements.
952
953    Args:
954        label: The cons operator.
955        kast: The cons list to flatten.
956
957    Returns:
958        Items of cons list.
959    """
960    flattened_args = []
961    rest_of_args = [kast]  # Rest of arguments in reversed order
962    while rest_of_args:
963        current_arg = rest_of_args.pop()
964        if isinstance(current_arg, KApply) and current_arg.label.name == label:
965            rest_of_args.extend(reversed(current_arg.args))
966        else:
967            flattened_args.append(current_arg)
968    return flattened_args
