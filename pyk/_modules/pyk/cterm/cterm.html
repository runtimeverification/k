 37class CTerm:
 38    """Represent a symbolic program state, obtained and manipulated using symbolic execution.
 39
 40    Contains the data:
 41    - `config`: the _configuration_ (structural component of the state, potentially containing free variabls)
 42    - `constraints`: conditions which limit/constraint the free variables from the `config`
 43    """
 44

 46    constraints: tuple[KInner, ...]

 48    def __init__(self, config: KInner, constraints: Iterable[KInner] = ()) -> None:
 49        """Instantiate a given `CTerm`, performing basic sanity checks on the `config` and `constraints`."""
 50        if is_top(config, weak=True):
 51            config = mlTop()
 52            constraints = ()
 53        elif is_bottom(config, weak=True):
 54            config = mlBottom()
 55            constraints = ()
 56        else:
 57            self._check_config(config)
 58            constraints = self._normalize_constraints(constraints)
 59        object.__setattr__(self, 'config', config)
 60        object.__setattr__(self, 'constraints', constraints)

 63    def from_kast(kast: KInner) -> CTerm:
 64        """Interpret a given `KInner` as a `CTerm` by splitting the `config` and `constraints` (see `CTerm.kast`)."""
 65        if is_top(kast, weak=True):
 66            return CTerm.top()
 67        elif is_bottom(kast, weak=True):
 68            return CTerm.bottom()
 69        else:
 70            config, constraint = split_config_and_constraints(kast)
 71            constraints = flatten_label('#And', constraint)
 72            return CTerm(config, constraints)

156    def match(self, cterm: CTerm) -> Subst | None:
157        """Find `Subst` instantiating this `CTerm` to the other, return `None` if no such `Subst` exists."""
158        csubst = self.match_with_constraint(cterm)
159
160        if not csubst:
161            return None
162
163        if csubst.constraint != mlTop(GENERATED_TOP_CELL):
164            return None
165
166        return csubst.subst

168    def match_with_constraint(self, cterm: CTerm) -> CSubst | None:
169        """Find `CSubst` instantiating this `CTerm` to the other, return `None` if no such `CSubst` exists."""
170        subst = self.config.match(cterm.config)
171
172        if subst is None:
173            return None
174
175        source_constraints = [subst(c) for c in self.constraints]
176        constraints = [c for c in cterm.constraints if c not in source_constraints]
177
178        return CSubst(subst, constraints)

180    @staticmethod
181    def _ml_impl(antecedents: Iterable[KInner], consequents: Iterable[KInner]) -> KInner:
182        antecedent = mlAnd(unique(antecedents), GENERATED_TOP_CELL)
183        consequent = mlAnd(unique(term for term in consequents if term not in set(antecedents)), GENERATED_TOP_CELL)
184
185        if mlTop(GENERATED_TOP_CELL) in {antecedent, consequent}:
186            return consequent
187
188        return mlImplies(antecedent, consequent, GENERATED_TOP_CELL)

194    def anti_unify(
195        self, other: CTerm, keep_values: bool = False, kdef: KDefinition | None = None
196    ) -> tuple[CTerm, CSubst, CSubst]:
197        """Given two `CTerm` instances, find a more general `CTerm` which can instantiate to both.
198
199        Args:
200            other: other `CTerm` to consider for finding a more general `CTerm` with this one.
201            keep_values: do not discard information about abstracted variables in returned result.
202            kdef (optional): `KDefinition` to make analysis more precise.
203
204        Returns:
205            A tuple ``(cterm, csubst1, csubst2)`` where
206
207            - ``cterm``: More general `CTerm` than either `self` or `other`.
208            - ``csubst1``: Constrained substitution to apply to `cterm` to obtain `self`.
209            - ``csubst2``: Constrained substitution to apply to `cterm` to obtain `other`.
210        """
211        new_config, self_subst, other_subst = anti_unify(self.config, other.config, kdef=kdef)
212        common_constraints = [constraint for constraint in self.constraints if constraint in other.constraints]
213
214        new_cterm = CTerm(config=new_config, constraints=())
215        if keep_values:
216            # todo: It's not able to distinguish between constraints in different cterms,
217            #  because variable names may be used inconsistently in different cterms.
218            self_unique_constraints = [
219                ml_pred_to_bool(constraint) for constraint in self.constraints if constraint not in other.constraints
220            ]
221            other_unique_constraints = [
222                ml_pred_to_bool(constraint) for constraint in other.constraints if constraint not in self.constraints
223            ]
224            disjunct_lhs = andBool([self_subst.pred] + self_unique_constraints)
225            disjunct_rhs = andBool([other_subst.pred] + other_unique_constraints)
226            if KToken('true', 'Bool') not in [disjunct_lhs, disjunct_rhs]:
227                new_cterm = new_cterm.add_constraint(mlEqualsTrue(orBool([disjunct_lhs, disjunct_rhs])))
228
229        new_constraints = remove_useless_constraints(common_constraints, new_cterm.free_vars)
230
231        for constraint in new_constraints:
232            new_cterm = new_cterm.add_constraint(constraint)
233        self_csubst = new_cterm.match_with_constraint(self)
234        other_csubst = new_cterm.match_with_constraint(other)
235        if self_csubst is None or other_csubst is None:
236            raise ValueError(
237                f'Anti-unification failed to produce a more general state: {(new_cterm, (self, self_csubst), (other, other_csubst))}'
238            )
239        return (new_cterm, self_csubst, other_csubst)

241    def remove_useless_constraints(self, keep_vars: Iterable[str] = ()) -> CTerm:
242        """Return a new `CTerm` with constraints over unbound variables removed.
243
244        Args:
245            keep_vars: List of variables to keep constraints for even if unbound in the `CTerm`.
246
247        Returns:
248            A `CTerm` with the constraints over unbound variables removed.
249        """
250        initial_vars = free_vars(self.config) | set(keep_vars)
251        new_constraints = remove_useless_constraints(self.constraints, initial_vars)
252        return CTerm(self.config, new_constraints)

255def anti_unify(state1: KInner, state2: KInner, kdef: KDefinition | None = None) -> tuple[KInner, Subst, Subst]:
256    """Return a generalized state over the two input states.
257
258    Args:
259        state1: State to generalize over, represented as bare `KInner`.
260        state2: State to generalize over, represented as bare `KInner`.
261        kdef (optional): `KDefinition` to make the analysis more precise.
262
263    Note:
264        Both `state1` and `state2` are expected to be bare configurations with no constraints attached.
265
266    Returns:
267        A tuple ``(state, subst1, subst2)`` such that
268
269        - ``state``: A symbolic state represented as `KInner` which is more general than `state1` or `state2`.
270        - ``subst1``: A `Subst` which, when applied to `state`, recovers `state1`.
271        - ``subst2``: A `Subst` which, when applied to `state`, recovers `state2`.
272    """
273
274    def _rewrites_to_abstractions(_kast: KInner) -> KInner:
275        if type(_kast) is KRewrite:
276            sort = kdef.sort(_kast) if kdef else None
277            return abstract_term_safely(_kast, sort=sort)
278        return _kast
279
280    minimized_rewrite = push_down_rewrites(KRewrite(state1, state2))
281    abstracted_state = bottom_up(_rewrites_to_abstractions, minimized_rewrite)
282    subst1 = abstracted_state.match(state1)
283    subst2 = abstracted_state.match(state2)
284    if subst1 is None or subst2 is None:
285        raise ValueError('Anti-unification failed to produce a more general state!')
286    return (abstracted_state, subst1, subst2)

330    def pred(self, sort_with: KDefinition | None = None, subst: bool = True, constraints: bool = True) -> KInner:
331        """Return an ML predicate representing this substitution."""
332        _preds: list[KInner] = []
333        if subst:
334            for k, v in self.subst.minimize().items():
335                sort = K
336                if sort_with is not None:
337                    _sort = sort_with.sort(v)
338                    sort = _sort if _sort is not None else sort
339                _preds.append(mlEquals(KVariable(k, sort=sort), v, arg_sort=sort))
340        if constraints:
341            _preds.extend(self.constraints)
342        return mlAnd(_preds)

428def cterms_anti_unify(
429    cterms: Iterable[CTerm], keep_values: bool = False, kdef: KDefinition | None = None
430) -> tuple[CTerm, list[CSubst]]:
431    """Given many `CTerm` instances, find a more general `CTerm` which can instantiate to all.
432
433    Args:
434        cterms: `CTerm`s to consider for finding a more general `CTerm` with this one.
435        keep_values: do not discard information about abstracted variables in returned result.
436        kdef (optional): `KDefinition` to make analysis more precise.
437
438    Returns:
439        A tuple ``(cterm, csubsts)`` where
440
441        - ``cterm``: More general `CTerm` than any of the input `CTerm`s.
442        - ``csubsts``: List of `CSubst` which, when applied to `cterm`, yield the input `CTerm`s.
443    """
444    # TODO: optimize this function, reduce useless auto-generated variables.
445    cterms = list(cterms)
446    if not cterms:
447        raise ValueError('Anti-unification failed, no CTerms provided')
448    merged_cterm = cterms[0]
449    for cterm in cterms[1:]:
450        merged_cterm = merged_cterm.anti_unify(cterm, keep_values, kdef)[0]
451    csubsts = [not_none(merged_cterm.match_with_constraint(cterm)) for cterm in cterms]
452    return merged_cterm, csubsts
